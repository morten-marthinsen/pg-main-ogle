```python
def score_rsf(candidate, expectation_schema, latent_resonance):
    """
    RSF scoring is REQUIRED for all Big Idea candidates.
    """
    # Schema Distance — How far from what audience expects?
    schema_distance = calculate_schema_distance(
        candidate['big_idea_statement'],
        expectation_schema['saturated_claims'],
        expectation_schema['exhausted_metaphors']
    )

    # Resonance Depth — Does it connect to latent emotions?
    resonance_depth = calculate_resonance_depth(
        candidate['big_idea_statement'],
        latent_resonance['unnamed_emotions'],
        latent_resonance['identity_tensions']
    )

    # Resolution Accessibility — Can they "get it" in 3 seconds?
    resolution_access = assess_resolution_time(
        candidate['big_idea_statement'],
        candidate['components']['creative_wrapper']
    )

    # Temporal Fit — Right message for right now?
    temporal_fit = assess_temporal_alignment(
        candidate,
        latent_resonance['cultural_timing']
    )

    # RSF Composite
    rsf_composite = (schema_distance * resonance_depth * resolution_access * temporal_fit) / 1000

    return {
        'schema_distance': schema_distance,
        'resonance_depth': resonance_depth,
        'resolution_accessibility': resolution_access,
        'temporal_fit': temporal_fit,
        'rsf_composite': rsf_composite,
        'passes_threshold': rsf_composite >= 3.0
    }
```

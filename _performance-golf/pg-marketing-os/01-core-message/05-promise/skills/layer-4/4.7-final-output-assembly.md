# 4.7 Final Output Assembly

## Purpose
Assemble all Promise skill outputs into a complete, deployment-ready package. This is the final gate of the Promise skill—producing everything the copywriter needs to write promise-driven copy.

---

## Input Requirements

From All Previous Layers:

Layer 1 (Generation):
- Raw promise candidates
- Promise types and frames

Layer 2 (Calibration):
- Calibrated promises with scores
- Campaign thesis
- Top candidate rankings

Layer 3 (Validation):
- Validated promises
- Validation scores
- Issues and recommendations

Layer 4 (Output Packaging):
- Primary promise (4.1)
- Supporting promises (4.2)
- Promise variations (4.3)
- Usage matrix (4.4)
- Proof pairings (4.5)
- Copy integration guide (4.6)

---

## Execution Protocol

### Step 1: Compile Executive Summary

```python
def compile_executive_summary(all_outputs):
    """
    Create quick-reference executive summary.
    """
    primary = all_outputs['primary_promise']
    thesis = all_outputs['campaign_thesis']
    supporting = all_outputs['supporting_promises']
    validation = all_outputs['validation_assembly']

    summary = {
        'project_name': all_outputs.get('project_name', 'Promise Package'),
        'generated_date': datetime.now().isoformat(),

        'primary_promise': {
            'statement': primary['statement'],
            'short_form': primary['variations'].get('headline', {}).get('text', primary['statement'][:50] + '...'),
            'validation_status': primary['validation_status']['status'],
            'proof_level': 'KNOCKOUT' if primary.get('has_knockout_proof') else 'ADEQUATE'
        },

        'campaign_thesis': {
            'statement': thesis['primary_thesis']['statement'],
            'score': thesis['primary_thesis']['scores']['composite']
        },

        'supporting_promises': {
            'count': len(supporting),
            'names': [sp['role'] for sp in supporting],
            'all_validated': all(sp['validation_status']['status'] != 'VALIDATION_FAILED' for sp in supporting)
        },

        'key_metrics': {
            'total_promises_generated': validation['summary']['total_validated'],
            'promises_fully_validated': validation['summary']['status_distribution']['fully_validated'],
            'average_validation_score': round(sum(
                validation['summary']['dimension_averages'].values()
            ) / len(validation['summary']['dimension_averages']), 2),
            'positions_covered': len(all_outputs['usage_matrix']['matrix'])
        },

        'proof_highlights': {
            'knockout_proof_count': all_outputs['proof_pairing']['proof_coverage_validation']['knockout_count'],
            'primary_proof_type': all_outputs['proof_pairing']['primary_promise_proof']['primary_proof']['proof_type'],
            'primary_proof_summary': all_outputs['proof_pairing']['primary_promise_proof']['primary_proof']['content'][:100] + '...'
        },

        'compliance_status': {
            'disclaimers_required': len(validation['summary'].get('disclaimers_needed', [])),
            'disclaimer_list': validation['summary'].get('disclaimers_needed', [])
        },

        'ready_for_copy': True
    }

    return summary
```

### Step 2: Assemble Primary Promise Package

```python
def assemble_primary_promise_package(primary_promise, variations, proof_pairing, usage_matrix, integration_guide):
    """
    Create complete primary promise package.
    """
    package = {
        'promise_id': primary_promise['promise_id'],
        'tier': 'PRIMARY',

        'core_statement': {
            'full': primary_promise['statement'],
            'core': extract_core_promise(primary_promise['statement']),
            'word_count': len(primary_promise['statement'].split())
        },

        'variations_by_type': {},
        'variations_by_position': {},

        'validation': {
            'status': primary_promise['validation_status']['status'],
            'composite_score': primary_promise['validation_composite'],
            'dimension_scores': primary_promise['validation_scores'],
            'issues': primary_promise.get('issues', [])
        },

        'proof_package': {
            'primary_proof': proof_pairing['primary_promise_proof']['primary_proof'],
            'position_proof': proof_pairing['primary_promise_proof']['position_proof'],
            'backup_proof': proof_pairing['primary_promise_proof']['backup_proof']
        },

        'usage_guidance': {
            'assigned_positions': list(usage_matrix['matrix'].keys()),
            'primary_positions': usage_matrix['position_summary']['primary_promise_positions'],
            'position_instructions': {}
        },

        'integration': {
            'section_instructions': {},
            'examples': {},
            'warnings': [],
            'checklist_items': []
        },

        'compliance': {
            'disclaimers_required': primary_promise.get('disclaimers_required', []),
            'platform_restrictions': primary_promise.get('platform_restrictions', []),
            'claim_qualifications': primary_promise.get('claim_qualifications', [])
        }
    }

    # Organize variations by type
    for var_type, var_list in variations.items():
        if var_list:
            package['variations_by_type'][var_type] = [
                {'text': v['text'], 'score': v.get('score', 5)}
                for v in var_list
            ]

    # Organize variations by position
    for position in usage_matrix['position_summary']['primary_promise_positions']:
        position_data = usage_matrix['matrix'].get(position, {})
        if position_data:
            package['variations_by_position'][position] = {
                'recommended_variation': position_data.get('primary_choice', {}).get('text'),
                'variation_type': position_data.get('primary_choice', {}).get('variation_type'),
                'alternatives': [alt.get('text') for alt in position_data.get('alternatives', [])]
            }

    # Add integration instructions
    for section, instructions in integration_guide['section_instructions'].items():
        if instructions.get('promise_to_use') and primary_promise['statement'] in instructions['promise_to_use']:
            package['integration']['section_instructions'][section] = instructions
            if instructions.get('example'):
                package['integration']['examples'][section] = instructions['example']
            if instructions.get('warnings'):
                package['integration']['warnings'].extend(instructions['warnings'])

    return package
```

### Step 3: Assemble Supporting Promise Packages

```python
def assemble_supporting_promise_packages(supporting_promises, all_variations, proof_pairings, usage_matrix):
    """
    Create packages for each supporting promise.
    """
    packages = []

    for sp in supporting_promises:
        promise_id = sp['promise_id']

        package = {
            'promise_id': promise_id,
            'tier': 'SUPPORTING',
            'role': sp['role'],
            'role_description': get_role_description(sp['role']),

            'core_statement': {
                'full': sp['statement'],
                'core': extract_core_promise(sp['statement'])
            },

            'why_selected': sp.get('selection_rationale', ''),

            'variations': all_variations.get(promise_id, {}),

            'validation': {
                'status': sp['validation_status']['status'],
                'score': sp.get('validation_composite', 5)
            },

            'proof': proof_pairings.get('supporting_promise_proof', []),

            'usage': {
                'assigned_positions': sp.get('usage_positions', []),
                'best_positions': sp.get('best_fit_positions', [])
            },

            'relationship_to_primary': sp.get('primary_connection', {})
        }

        packages.append(package)

    return packages


def get_role_description(role):
    """Get description of supporting promise role."""
    descriptions = {
        'benefit_expansion': 'Expands value perception beyond primary benefit',
        'objection_preemption': 'Addresses likely objections before they arise',
        'emotional_variety': 'Provides emotional range to connect with different motivations',
        'proof_anchor': 'Strong proof attached; anchors credibility',
        'mechanism_reinforcement': 'Reinforces mechanism story coherence'
    }
    return descriptions.get(role, 'Supporting role')
```

### Step 4: Generate Deployment Checklist

```python
def generate_deployment_checklist(full_package):
    """
    Create comprehensive deployment checklist.
    """
    checklist = {
        'before_writing': [],
        'during_writing': [],
        'after_writing': [],
        'before_publishing': []
    }

    # Before writing
    checklist['before_writing'] = [
        {
            'item': 'Review primary promise and ensure understanding',
            'reference': 'primary_promise_package.core_statement'
        },
        {
            'item': 'Review all variations available',
            'reference': 'primary_promise_package.variations_by_type'
        },
        {
            'item': 'Review proof pairings for each position',
            'reference': 'primary_promise_package.proof_package'
        },
        {
            'item': 'Note all compliance requirements',
            'reference': 'primary_promise_package.compliance'
        },
        {
            'item': 'Review supporting promises and their roles',
            'reference': 'supporting_promise_packages'
        },
        {
            'item': 'Confirm copy format and section structure',
            'reference': 'copy_integration_guide.section_instructions'
        }
    ]

    # During writing - per section
    checklist['during_writing'] = []

    for section, inst in full_package['copy_integration_guide']['section_instructions'].items():
        checklist['during_writing'].append({
            'section': section,
            'items': [
                f"Use {inst.get('variation_type', 'appropriate')} variation",
                f"Deploy proof: {inst.get('proof_to_deploy', {}).get('type', 'as available')}",
                f"Target word count: {inst.get('word_count_target', 'varies')}",
                f"Follow format: {inst.get('integration_format', 'standard')}"
            ]
        })

    # After writing
    checklist['after_writing'] = [
        'Verify no exact promise repetition (variations used throughout)',
        'Verify proof appears within 100 words of each major claim',
        'Verify primary promise appears in opening AND close',
        'Verify supporting promises properly distributed',
        'Verify emotional progression from opening to close'
    ]

    # Before publishing
    checklist['before_publishing'] = [
        'All required disclaimers present',
        'Platform compliance verified',
        'Claim-proof alignment verified',
        'Testimonials have required disclosures',
        'Legal review complete (if required)'
    ]

    # Add specific compliance items
    disclaimers = full_package.get('executive_summary', {}).get('compliance_status', {}).get('disclaimer_list', [])
    for disc in disclaimers:
        checklist['before_publishing'].append(f"Include disclaimer: {disc}")

    return checklist
```

### Step 5: Assemble Final Package

```python
def assemble_final_package(all_outputs):
    """
    Assemble complete final output package.
    """
    package = {
        'package_version': '1.0',
        'generated': datetime.now().isoformat(),
        'skill': '05-promise',
        'layer': 4,
        'microskill': '4.7-final-output-assembly',

        # Executive Summary
        'executive_summary': compile_executive_summary(all_outputs),

        # Primary Promise Package
        'primary_promise_package': assemble_primary_promise_package(
            all_outputs['primary_promise'],
            all_outputs['variations'].get(all_outputs['primary_promise']['promise_id'], {}),
            all_outputs['proof_pairing'],
            all_outputs['usage_matrix'],
            all_outputs['integration_guide']
        ),

        # Supporting Promise Packages
        'supporting_promise_packages': assemble_supporting_promise_packages(
            all_outputs['supporting_promises'],
            all_outputs['variations'],
            all_outputs['proof_pairing'],
            all_outputs['usage_matrix']
        ),

        # Campaign Thesis Package
        'campaign_thesis_package': {
            'primary_thesis': all_outputs['campaign_thesis']['primary_thesis'],
            'runners_up': all_outputs['campaign_thesis']['runners_up'],
            'variations': all_outputs['campaign_thesis'].get('thesis_variations', {}),
            'competitive_positioning': all_outputs['campaign_thesis'].get('competitive_positioning', {})
        },

        # Copy Integration Guide
        'copy_integration_guide': all_outputs['integration_guide'],

        # Deployment Checklist
        'deployment_checklist': None,  # Generated below

        # Reference Appendices
        'appendices': {
            'full_validation_report': all_outputs['validation_assembly'],
            'proof_inventory_reference': all_outputs['proof_pairing']['proof_type_distribution'],
            'all_promise_variations': all_outputs['variations'],
            'usage_matrix_full': all_outputs['usage_matrix'],
            'testing_recommendations': generate_testing_recommendations(all_outputs)
        }
    }

    # Generate deployment checklist with full context
    package['deployment_checklist'] = generate_deployment_checklist(package)

    return package


def generate_testing_recommendations(all_outputs):
    """
    Generate A/B testing recommendations.
    """
    recommendations = {
        'headline_tests': [],
        'email_subject_tests': [],
        'ad_creative_tests': [],
        'thesis_tests': []
    }

    # Headline variations to test
    primary_variations = all_outputs['variations'].get(
        all_outputs['primary_promise']['promise_id'], {}
    )

    if primary_variations.get('headline'):
        recommendations['headline_tests'] = [
            {
                'control': primary_variations['headline'][0]['text'],
                'variant': v['text'],
                'hypothesis': f"Testing {v.get('angle', 'alternate angle')}"
            }
            for v in primary_variations['headline'][1:3]
        ]

    # Email subject tests
    if primary_variations.get('subject_line'):
        recommendations['email_subject_tests'] = [
            {'subject': v['text'], 'priority': i + 1}
            for i, v in enumerate(primary_variations['subject_line'][:5])
        ]

    # Thesis tests
    thesis_data = all_outputs['campaign_thesis']
    recommendations['thesis_tests'] = [
        {
            'thesis': thesis_data['primary_thesis']['statement'],
            'role': 'control'
        }
    ] + [
        {
            'thesis': ru['statement'],
            'role': 'variant',
            'use_when': ru.get('use_when')
        }
        for ru in thesis_data.get('runners_up', [])[:2]
    ]

    return recommendations
```

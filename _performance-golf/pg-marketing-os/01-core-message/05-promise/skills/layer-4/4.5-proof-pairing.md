# 4.5 Proof Pairing

## Purpose
Pair each promise with its optimal proof from the proof inventory. Every promise needs evidence—this microskill ensures the right proof hits at the right moment to maximize believability and conversion.

From 4.1-4.4:

- Supporting promises with variations
- Usage matrix (promise-to-position mapping)

- Proof types (testimonial, demonstration, statistic, expert, scientific, logical)
- Proof-to-claim mappings

From 04-mechanism:

## Execution Protocol

### Step 1: Map Available Proof to Promises

```python
def map_proof_to_promises(promises, proof_inventory):
    """
    Create initial mapping of available proof for each promise.
    """
    proof_map = {}

    for promise in promises:
        promise_id = promise['promise_id']
        promise_core = extract_core_promise(promise['statement'])

        proof_map[promise_id] = {
            'promise': promise['statement'],
            'available_proof': [],
            'proof_gap': False
        }

        for proof in proof_inventory:
            relevance = assess_proof_relevance(promise_core, proof)

            if relevance['score'] >= 6:  # Threshold for relevance
                proof_map[promise_id]['available_proof'].append({
                    'proof_id': proof['proof_id'],
                    'proof_type': proof['type'],
                    'proof_strength': proof['strength_score'],
                    'proof_content': proof['content'],
                    'relevance_score': relevance['score'],
                    'relevance_reason': relevance['reason']
                })

        # Sort by combined strength + relevance
        proof_map[promise_id]['available_proof'].sort(
            key=lambda x: (x['proof_strength'] * 0.6 + x['relevance_score'] * 0.4),
            reverse=True
        )

        # Flag gaps
        if not proof_map[promise_id]['available_proof']:
            proof_map[promise_id]['proof_gap'] = True

    return proof_map


def assess_proof_relevance(promise_core, proof):
    """
    Assess how relevant a proof element is to a promise.
    """
    relevance = {
        'score': 5,
        'reason': []
    }

    # Direct claim match
    if claims_match(promise_core, proof.get('claim_supported')):
        relevance['score'] += 3
        relevance['reason'].append('Direct claim match')

    # Outcome alignment
    if outcomes_align(promise_core, proof.get('outcome_demonstrated')):
        relevance['score'] += 2
        relevance['reason'].append('Outcome alignment')

    # Mechanism connection
    if proof.get('mechanism_connected'):
        relevance['score'] += 1
        relevance['reason'].append('Mechanism connected')

    # Specificity match
    promise_specificity = assess_specificity(promise_core)
    proof_specificity = proof.get('specificity_level', 5)
    if abs(promise_specificity - proof_specificity) <= 2:
        relevance['score'] += 1
        relevance['reason'].append('Specificity matched')

    return relevance
```

### Step 2: Select Primary Proof Per Promise

```python
def select_primary_proof(promise, available_proof, position=None):
    """
    Select the single best proof for a promise.
    """
    if not available_proof:
        return {
            'selected': None,
            'status': 'NO_PROOF_AVAILABLE',
            'recommendation': 'Flag for proof development or promise revision'
        }

    # Filter by position requirements if specified
    if position:
        position_preferences = get_position_proof_preferences(position)
        filtered = [p for p in available_proof
                    if p['proof_type'] in position_preferences.get('preferred_types', [])]
        if filtered:
            available_proof = filtered

    # Select highest combined score
    best_proof = available_proof[0]

    # Check if knockout level
    is_knockout = best_proof['proof_strength'] >= 8 and best_proof['relevance_score'] >= 8

    return {
        'selected': best_proof,
        'is_knockout': is_knockout,
        'status': 'KNOCKOUT' if is_knockout else 'ADEQUATE',
        'backup_options': available_proof[1:3] if len(available_proof) > 1 else []
    }
```

### Step 3: Build Proof Stack Per Promise

```python
def build_proof_stack(promise, available_proof, usage_positions):
    """
    Build a complete proof stack for a promise across positions.
    """
    proof_stack = {
        'promise_id': promise['promise_id'],
        'promise': promise['statement'],
        'primary_proof': None,
        'position_proof': {},
        'backup_proof': [],
        'total_proof_types_used': 0
    }

    # Select primary (knockout) proof
    primary_selection = select_primary_proof(promise, available_proof)
    proof_stack['primary_proof'] = primary_selection['selected']

    # Remove primary from pool for position assignments
    remaining_proof = [p for p in available_proof
                       if p != primary_selection['selected']]

    # Assign proof to each usage position
    used_proof_types = set()
    if proof_stack['primary_proof']:
        used_proof_types.add(proof_stack['primary_proof']['proof_type'])

    for position in usage_positions:
        position_selection = select_primary_proof(
            promise, remaining_proof, position
        )

        if position_selection['selected']:
            # Prefer variety of proof types
            if position_selection['selected']['proof_type'] not in used_proof_types:
                proof_stack['position_proof'][position] = position_selection['selected']
                used_proof_types.add(position_selection['selected']['proof_type'])
            else:
                # Use different proof of same type if available
                for alt in remaining_proof:
                    if alt['proof_id'] != position_selection['selected']['proof_id']:
                        proof_stack['position_proof'][position] = alt
                        break
                else:
                    proof_stack['position_proof'][position] = position_selection['selected']

    proof_stack['total_proof_types_used'] = len(used_proof_types)
    proof_stack['backup_proof'] = remaining_proof[:3]

    return proof_stack
```

```python
# Helper functions for deployment generation
def extract_quote_snippet(content, max_words):
    """Extract punchy snippet from testimonial."""
    quote = content.get('quote', '')
    words = quote.split()[:max_words]
    return ' '.join(words) + ('...' if len(quote.split()) > max_words else '')


def format_stat_punchy(content):
    """Format statistic for lead/hook."""
    percentage = content.get('percentage', content.get('result'))
    return f"{percentage}" if percentage else "Proven results"


def extract_emotional_quote(content):
    """Extract most emotional part of testimonial."""
    quote = content.get('quote', '')
    # Look for emotional language
    emotional_phrases = ['finally', 'never thought', 'changed my life', 'can\'t believe']
    for phrase in emotional_phrases:
        if phrase in quote.lower():
            # Extract sentence containing phrase
            sentences = quote.split('.')
            for s in sentences:
                if phrase in s.lower():
                    return s.strip()
    return extract_quote_snippet(content, 20)
```

### Step 5: Validate Proof Coverage

```python
def validate_proof_coverage(all_proof_stacks, promises):
    """
    Validate that all promises have adequate proof coverage.
    """
    validation = {
        'all_covered': True,
        'knockout_count': 0,
        'adequate_count': 0,
        'gaps': [],
        'recommendations': []
    }

    for promise in promises:
        promise_id = promise['promise_id']
        stack = all_proof_stacks.get(promise_id)

        if not stack or not stack['primary_proof']:
            validation['all_covered'] = False
            validation['gaps'].append({
                'promise_id': promise_id,
                'promise': promise['statement'],
                'gap_type': 'NO_PROOF'
            })
            continue

        primary_strength = stack['primary_proof']['proof_strength']

        if primary_strength >= 8:
            validation['knockout_count'] += 1
        elif primary_strength >= 5:
            validation['adequate_count'] += 1
        else:
            validation['gaps'].append({
                'promise_id': promise_id,
                'promise': promise['statement'],
                'gap_type': 'WEAK_PROOF',
                'current_strength': primary_strength
            })

    # Generate recommendations
    if validation['knockout_count'] == 0:
        validation['recommendations'].append(
            'No knockout proof for primary promise — consider strengthening proof or reducing claim'
        )

    for gap in validation['gaps']:
        if gap['gap_type'] == 'NO_PROOF':
            validation['recommendations'].append(
                f"Promise '{gap['promise'][:50]}...' has no supporting proof — recommend removal or proof development"
            )
        elif gap['gap_type'] == 'WEAK_PROOF':
            validation['recommendations'].append(
                f"Promise '{gap['promise'][:50]}...' has weak proof ({gap['current_strength']}/10) — consider downgrading claim"
            )

    return validation
```

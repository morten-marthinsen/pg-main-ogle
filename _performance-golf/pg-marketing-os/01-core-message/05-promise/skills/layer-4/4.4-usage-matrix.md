# 4.4 Usage Matrix

## Purpose
Map each promise and variation to optimal copy positions. Not every promise belongs everywhere—this matrix ensures the right promise hits the right spot in the copy.

---

## Input Requirements

From 4.1-4.3:
- Primary promise with variations
- Supporting promises with variations
- Variation scores

From 08-structure (if available):
- Copy outline
- Section requirements

From Context:
- Copy format (VSL, sales letter, email sequence)
- Copy length
- Platform

---

## Execution Protocol

### Step 1: Assess Position Fit

```python
def assess_position_fit(promise, position, variations):
    """
    Score how well a promise fits a specific position.
    """
    position_requirements = POSITION_REQUIREMENTS[position]
    scores = {}

    # Base fit from promise attributes
    base_score = 5

    # Check requirements
    for req in position_requirements.get('requirements', []):
        if check_requirement(promise, req):
            base_score += 1
        else:
            base_score -= 0.5

    # Check variation availability
    preferred_types = position_requirements.get('preferred_variations', [])
    available_variations = []

    for var_type in preferred_types:
        if var_type in variations and variations[var_type]:
            available_variations.extend(variations[var_type])
            base_score += 0.5

    # Check variation quality
    if available_variations:
        best_var_score = max(v['score'] for v in available_variations)
        base_score += (best_var_score - 5) / 2  # Boost for high-quality variations

    return {
        'position': position,
        'fit_score': min(10, max(1, base_score)),
        'best_variation': available_variations[0] if available_variations else None,
        'variation_options': available_variations[:3]
    }


POSITION_REQUIREMENTS = {
    'lead': {
        'requirements': ['high_emotional_impact', 'can_stand_alone'],
        'preferred_variations': ['hook', 'headline', 'question']
    },
    'thesis_statement': {
        'requirements': ['is_primary', 'mechanism_connected', 'differentiated'],
        'preferred_variations': ['statement']
    },
    'mechanism_bridge': {
        'requirements': ['mechanism_connected'],
        'preferred_variations': ['statement', 'question']
    },
    'proof_section': {
        'requirements': ['has_knockout_proof', 'specific'],
        'preferred_variations': ['testimonial_setup', 'statement']
    },
    'benefit_stack': {
        'requirements': [],  # All promises can be bullets
        'preferred_variations': ['benefit_bullet']
    },
    'close': {
        'requirements': ['high_emotional_resonance'],
        'preferred_variations': ['future_pace', 'statement']
    },
    'email_subject': {
        'requirements': ['short', 'curiosity_driven'],
        'preferred_variations': ['subject_line', 'hook']
    },
    'ad_creative': {
        'requirements': ['platform_compliant', 'attention_grabbing'],
        'preferred_variations': ['ad_copy', 'hook', 'headline']
    }
}


def check_requirement(promise, requirement):
    """Check if promise meets a specific requirement."""
    checks = {
        'high_emotional_impact': lambda p: p.get('emotional_resonance_score', 5) >= 7,
        'can_stand_alone': lambda p: True,  # Most can
        'is_primary': lambda p: p.get('tier') == 'PRIMARY',
        'mechanism_connected': lambda p: p.get('mechanism_connected', False),
        'differentiated': lambda p: p.get('differentiation_score', 5) >= 7,
        'has_knockout_proof': lambda p: p.get('proof_verification', 5) >= 8,
        'specific': lambda p: p.get('specificity_score', 5) >= 7,
        'high_emotional_resonance': lambda p: p.get('emotional_resonance_score', 5) >= 7,
        'short': lambda p: len(p.get('statement', '').split()) <= 15,
        'curiosity_driven': lambda p: True,
        'platform_compliant': lambda p: p.get('compliance_score', 5) >= 6,
        'attention_grabbing': lambda p: p.get('emotional_impact', 5) >= 6
    }
    return checks.get(requirement, lambda p: True)(promise)
```

### Step 2: Build Usage Matrix

```python
def build_usage_matrix(promises, variations_by_promise):
    """
    Build complete usage matrix mapping promises to positions.
    """
    positions = list(POSITION_REQUIREMENTS.keys())
    matrix = {pos: [] for pos in positions}

    for promise in promises:
        promise_id = promise['promise_id']
        variations = variations_by_promise.get(promise_id, {})

        for position in positions:
            fit = assess_position_fit(promise, position, variations)

            if fit['fit_score'] >= 5:  # Minimum threshold
                matrix[position].append({
                    'promise_id': promise_id,
                    'promise_tier': promise.get('tier'),
                    'fit_score': fit['fit_score'],
                    'best_variation': fit['best_variation'],
                    'alternatives': fit['variation_options']
                })

    # Sort each position by fit score
    for position in matrix:
        matrix[position].sort(key=lambda x: x['fit_score'], reverse=True)

    return matrix


def generate_usage_recommendations(matrix):
    """
    Generate specific recommendations for each position.
    """
    recommendations = {}

    for position, candidates in matrix.items():
        if not candidates:
            recommendations[position] = {
                'status': 'NO_FIT',
                'recommendation': 'Consider generating new variations for this position'
            }
            continue

        primary = candidates[0]
        alternatives = candidates[1:3]

        recommendations[position] = {
            'status': 'READY',
            'primary_choice': {
                'promise_id': primary['promise_id'],
                'variation': primary['best_variation'],
                'fit_score': primary['fit_score']
            },
            'alternatives': [
                {
                    'promise_id': alt['promise_id'],
                    'variation': alt['best_variation']
                } for alt in alternatives
            ],
            'usage_note': generate_position_note(position, primary)
        }

    return recommendations


def generate_position_note(position, choice):
    """Generate specific usage note for position."""
    notes = {
        'lead': f"Lead with {choice['best_variation']['text'][:50]}... to hook attention",
        'thesis_statement': "Use full statement as core campaign declaration",
        'mechanism_bridge': "Transition: 'Here's how [variation] works...'",
        'proof_section': "Set up testimonial with this promise, then deliver proof",
        'benefit_stack': "Include in bullet stack with other benefits",
        'close': "Use for final emotional push before CTA",
        'email_subject': "A/B test top 3 subject line variations",
        'ad_creative': "Test this hook against alternatives"
    }
    return notes.get(position, "Use as recommended")
```

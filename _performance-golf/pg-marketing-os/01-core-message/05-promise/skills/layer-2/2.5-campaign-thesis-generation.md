# 2.5 Campaign Thesis Generation

## Purpose
Generate and refine campaign thesis statements that synthesize mechanism + promise. The campaign thesis is the single most important strategic statement in the copy—it's the core argument you're making.

---

From 2.1-2.4 (Calibrated Promises):

From 01-research:

- competitor_thesis_gaps

---

## Campaign Thesis Formula

```
PRIMARY FORMULA (E5 Method):
"The [SUPERIORITY] way to [PROMISE] is with the [MECHANISM]"

COMPONENTS:
- SUPERIORITY: What makes this approach better/different (fastest, only, easiest, etc.)
- PROMISE: The primary benefit/result (proof-ceiling compliant)
- MECHANISM: The unique pathway/solution (named mechanism)
```

## Execution Protocol

### Step 1: Identify Thesis Components

```python
def identify_thesis_components(calibrated_promises, mechanism_data, research_data):
    """
    Extract best components for thesis generation.
    """
    components = {
        'promises': [],
        'superiority_claims': [],
        'mechanism': None,
        'villain': None,
        'identity': None
    }

    # Best promises (differentiated + mechanism-verified)
    for promise in calibrated_promises:
        if promise.get('differentiation_score', 0) >= 7:
            if promise.get('mechanism_fit', {}).get('alignment_strength', 0) >= 7:
                components['promises'].append({
                    'statement': extract_core_promise(promise['statement']),
                    'score': promise.get('composite_score', 5),
                    'type': promise.get('promise_type')
                })

    # Sort by score
    components['promises'].sort(key=lambda x: x['score'], reverse=True)

    # Mechanism
    components['mechanism'] = {
        'name': mechanism_data['mechanism_name'],
        'type': mechanism_data.get('mechanism_type'),
        'superiority': mechanism_data.get('superiority_claim')
    }

    # Superiority claims
    if mechanism_data.get('superiority_claim'):
        components['superiority_claims'].append(mechanism_data['superiority_claim'])

    # Generate additional superiority options
    superiority_options = [
        'only', 'fastest', 'easiest', 'most effective',
        'most natural', 'safest', 'most reliable', 'proven'
    ]
    components['superiority_claims'].extend(superiority_options)

    # Villain
    if mechanism_data.get('villain'):
        components['villain'] = mechanism_data['villain']

    # Target identity (from research)
    if research_data.get('avatar', {}).get('identity_markers'):
        components['identity'] = research_data['avatar']['identity_markers'][0]

    return components
```

### Step 2: Generate Thesis Candidates

```python
def generate_thesis_candidates(components, schwartz_stage):
    """
    Generate multiple thesis candidates.
    """
    candidates = []
    mechanism_name = components['mechanism']['name']
    top_promises = components['promises'][:3]
    superiority_claims = components['superiority_claims']
    villain = components.get('villain')
    identity = components.get('identity')

    # Get appropriate templates for stage
    templates = get_stage_templates(schwartz_stage)

    for promise in top_promises:
        promise_core = promise['statement']

        # Generate from templates
        for template in templates:
            for superiority in superiority_claims[:3]:
                thesis = template.format(
                    superiority=superiority,
                    promise=promise_core,
                    mechanism=mechanism_name,
                    villain=villain or '',
                    identity=identity or ''
                )

                candidates.append({
                    'thesis': clean_thesis(thesis),
                    'promise_used': promise_core,
                    'superiority_used': superiority,
                    'template_type': get_template_type(template),
                    'stage_fit': schwartz_stage
                })

        # Special villain thesis (Stage 4)
        if villain and schwartz_stage >= 4:
            candidates.append({
                'thesis': f"The ONLY way to {promise_core} is to defeat the {villain} with the {mechanism_name}",
                'promise_used': promise_core,
                'superiority_used': 'only',
                'template_type': 'villain_defeat',
                'stage_fit': 4
            })

        # Identity thesis (Stage 5)
        if identity and schwartz_stage >= 5:
            candidates.append({
                'thesis': f"For {identity} who've tried everything: the {mechanism_name} way to finally {promise_core}",
                'promise_used': promise_core,
                'superiority_used': 'finally',
                'template_type': 'identity_led',
                'stage_fit': 5
            })

    return candidates


def clean_thesis(thesis):
    """Clean up thesis statement."""
    # Remove double spaces
    thesis = ' '.join(thesis.split())
    # Capitalize appropriately
    thesis = thesis[0].upper() + thesis[1:]
    return thesis
```

### Step 3: Score Thesis Candidates

```python
def score_thesis(candidate, context):
    """
    Score thesis on multiple dimensions.
    """
    scores = {}

    thesis = candidate['thesis']

    # CLARITY (1-10): Is it immediately understandable?
    scores['clarity'] = assess_clarity(thesis)

    # SPECIFICITY (1-10): Does it make a specific claim?
    scores['specificity'] = assess_specificity(thesis)

    # DIFFERENTIATION (1-10): Does it stand apart?
    scores['differentiation'] = candidate.get('differentiation_score', 7)

    # MECHANISM PROMINENCE (1-10): Is the mechanism clearly featured?
    if candidate.get('mechanism') in thesis:
        scores['mechanism_prominence'] = 9
    else:
        scores['mechanism_prominence'] = 5

    # STAGE FIT (1-10): Does it match the Schwartz stage?
    if candidate['stage_fit'] == context['schwartz_stage']:
        scores['stage_fit'] = 10
    elif abs(candidate['stage_fit'] - context['schwartz_stage']) == 1:
        scores['stage_fit'] = 7
    else:
        scores['stage_fit'] = 4

    # EMOTIONAL IMPACT (1-10): Does it create desire?
    scores['emotional_impact'] = assess_emotional_impact(thesis)

    # PROOF ALIGNMENT (1-10): Can we prove this?
    scores['proof_alignment'] = context.get('proof_ceiling_score', 7)

    # COMPOSITE
    weights = {
        'clarity': 0.15,
        'specificity': 0.15,
        'differentiation': 0.20,
        'mechanism_prominence': 0.15,
        'stage_fit': 0.15,
        'emotional_impact': 0.10,
        'proof_alignment': 0.10
    }

    composite = sum(scores[dim] * weights[dim] for dim in weights)
    scores['composite'] = round(composite, 2)

    return scores


def assess_clarity(thesis):
    """Assess thesis clarity."""
    # Shorter = clearer (generally)
    word_count = len(thesis.split())
    if word_count <= 15:
        base_score = 9
    elif word_count <= 20:
        base_score = 7
    else:
        base_score = 5

    # Deduct for jargon
    jargon_words = ['leverage', 'synergy', 'optimize', 'paradigm']
    for word in jargon_words:
        if word in thesis.lower():
            base_score -= 1

    return max(1, base_score)


def assess_emotional_impact(thesis):
    """Assess emotional resonance of thesis."""
    emotional_triggers = {
        'finally': 2,
        'only': 1.5,
        'never again': 2,
        'end': 1,
        'freedom': 1.5,
        'control': 1,
        'defeat': 1.5,
        'eliminate': 1
    }

    score = 5
    thesis_lower = thesis.lower()

    for trigger, boost in emotional_triggers.items():
        if trigger in thesis_lower:
            score += boost

    return min(10, score)
```

### Step 4: Select Primary Thesis

```python
def select_primary_thesis(candidates, context):
    """
    Select the primary campaign thesis.
    """
    scored_candidates = []

    for candidate in candidates:
        scores = score_thesis(candidate, context)
        candidate['scores'] = scores
        scored_candidates.append(candidate)

    # Sort by composite
    scored_candidates.sort(key=lambda x: x['scores']['composite'], reverse=True)

    primary = scored_candidates[0]
    runners_up = scored_candidates[1:4]

    return {
        'primary_thesis': primary,
        'runners_up': runners_up,
        'selection_rationale': generate_selection_rationale(primary, context)
    }


def generate_selection_rationale(thesis, context):
    """Explain why this thesis was selected."""
    scores = thesis['scores']
    rationale = []

    if scores['differentiation'] >= 8:
        rationale.append("Highly differentiated from competitors")

    if scores['mechanism_prominence'] >= 8:
        rationale.append("Strong mechanism positioning")

    if scores['emotional_impact'] >= 7:
        rationale.append("High emotional resonance")

    if scores['proof_alignment'] >= 8:
        rationale.append("Well-supported by proof inventory")

    return ' | '.join(rationale) if rationale else "Best overall composite score"
```

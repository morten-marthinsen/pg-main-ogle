# 2.1 Proof Ceiling Application

## Purpose
Apply proof ceiling constraints to promise candidates. The promise can NEVER exceed what proof can support. This is the primary constraint that keeps promises credible and provable.

**Core Principle:** The proof ceiling from 02-proof-inventory sets the maximum claim level. No amount of good copywriting can make an unprovable promise believable.

---

From 02-proof-inventory:
- promise_ceiling (enum: transformation, significant_improvement, noticeable_improvement, some_benefit, minimal)
- overall_strength_score (0-100)
- believability_constraints

## Execution Protocol

### Step 1: Classify Promise Claim Level

```python
def classify_claim_level(promise):
    """
    Determine what ceiling level a promise requires.
    """
    claim_indicators = {
        'transformation': [
            'transform', 'complete', 'total', 'permanent', 'life-changing',
            'never again', 'once and for all', 'forever', 'completely'
        ],
        'significant_improvement': [
            'significant', 'major', 'substantial', 'lasting', 'real',
            'dramatic', 'noticeable', 'measurable'
        ],
        'noticeable_improvement': [
            'improve', 'better', 'change', 'help', 'support',
            'positive', 'meaningful'
        ],
        'some_benefit': [
            'may', 'could', 'might', 'designed to', 'intended to',
            'some', 'possible'
        ]
    }

    promise_lower = promise.lower()

    # Check from highest to lowest
    for level, indicators in claim_indicators.items():
        if any(ind in promise_lower for ind in indicators):
            return level

    # Check specificity indicators
    if has_specific_numbers(promise):
        return 'significant_improvement'

    if has_timeframe_claim(promise):
        return 'noticeable_improvement'

    return 'some_benefit'  # Default conservative


def has_specific_numbers(promise):
    """Check for specific number claims."""
    import re
    patterns = [
        r'\d+\s*(pounds?|lbs?|%|days?|weeks?|hours?)',
        r'\d+x',
        r'\$\d+'
    ]
    return any(re.search(p, promise, re.I) for p in patterns)
```

### Step 2: Check Ceiling Compliance

```python
def check_ceiling_compliance(promise, required_level, actual_ceiling):
    """
    Check if promise is within proof ceiling.
    """
    level_order = ['minimal', 'some_benefit', 'noticeable_improvement',
                   'significant_improvement', 'transformation']

    required_index = level_order.index(required_level)
    ceiling_index = level_order.index(actual_ceiling)

    compliance = {
        'promise': promise,
        'required_level': required_level,
        'actual_ceiling': actual_ceiling,
        'compliant': required_index <= ceiling_index,
        'gap': required_index - ceiling_index if required_index > ceiling_index else 0
    }

    if not compliance['compliant']:
        compliance['exceeds_by'] = compliance['gap']
        compliance['action_required'] = 'downgrade'
        compliance['downgrade_to'] = actual_ceiling

    return compliance
```

### Step 3: Downgrade Non-Compliant Promises

```python
def downgrade_promise(promise, from_level, to_level):
    """
    Reduce promise strength to meet proof ceiling.
    """
    downgrade_map = {
        # Transformation → Significant
        ('transformation', 'significant_improvement'): {
            'complete transformation': 'significant improvement',
            'total change': 'major change',
            'permanent': 'lasting',
            'never again': 'dramatically reduce',
            'once and for all': 'for good',
            'life-changing': 'significant'
        },
        # Significant → Noticeable
        ('significant_improvement', 'noticeable_improvement'): {
            'significant': 'noticeable',
            'major': 'real',
            'substantial': 'meaningful',
            'lasting': 'positive',
            'dramatic': 'clear'
        },
        # Noticeable → Some Benefit
        ('noticeable_improvement', 'some_benefit'): {
            'noticeable': 'may help with',
            'improve': 'can support',
            'change': 'may improve',
            'real': 'some',
            'clear': 'potential'
        }
    }

    key = (from_level, to_level)
    replacements = downgrade_map.get(key, {})

    downgraded = promise
    replacements_made = []

    for original, replacement in replacements.items():
        if original.lower() in downgraded.lower():
            downgraded = downgraded.replace(original, replacement)
            downgraded = downgraded.replace(original.capitalize(), replacement.capitalize())
            replacements_made.append({
                'from': original,
                'to': replacement
            })

    # Handle specificity if needed
    if to_level in ['some_benefit', 'minimal']:
        downgraded = remove_specific_numbers(downgraded)
        replacements_made.append({'from': 'specific numbers', 'to': 'removed'})

    return {
        'original': promise,
        'downgraded': downgraded,
        'from_level': from_level,
        'to_level': to_level,
        'replacements': replacements_made
    }


def remove_specific_numbers(promise):
    """Remove specific numbers for heavily hedged promises."""
    import re

    # Replace specific numbers with general language
    patterns_replacements = [
        (r'\d+\s*pounds?', 'weight'),
        (r'\d+\s*%', 'noticeable'),
        (r'in\s*\d+\s*days?', 'over time'),
        (r'in\s*\d+\s*weeks?', 'with continued use'),
        (r'within\s*\d+\s*hours?', 'soon')
    ]

    result = promise
    for pattern, replacement in patterns_replacements:
        result = re.sub(pattern, replacement, result, flags=re.I)

    return result
```

### Step 4: Verify Specific Claim Proof

```python
def verify_specific_claims(promise, proof_inventory):
    """
    For promises with specific claims, verify proof exists.
    """
    specific_claims = extract_specific_claims(promise)
    verification = []

    for claim in specific_claims:
        proof_match = find_proof_for_claim(claim, proof_inventory)

        verification.append({
            'claim': claim,
            'proof_found': proof_match is not None,
            'proof_element': proof_match.get('element_id') if proof_match else None,
            'proof_strength': proof_match.get('strength') if proof_match else 0
        })

    all_verified = all(v['proof_found'] for v in verification)

    return {
        'all_claims_verified': all_verified,
        'verification_details': verification,
        'unverified_claims': [v['claim'] for v in verification if not v['proof_found']]
    }
```

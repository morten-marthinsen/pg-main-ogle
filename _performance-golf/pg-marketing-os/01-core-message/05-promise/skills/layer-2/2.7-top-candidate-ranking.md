# 2.7 Top Candidate Ranking

## Purpose
Rank all calibrated promises and select top candidates for Layer 3 validation. This is the output gate for Layer 2—producing a prioritized shortlist of promises that have passed all calibration checks.

---

## Input Requirements

From 2.6 Calibration Scoring:
- All candidates with calibration composite scores
- Tier assignments
- Dimension scores
- Calibration status

From 2.5 Campaign Thesis Generation:
- Primary thesis
- Thesis-supporting promises

From Context:
- Copy format requirements (VSL, sales letter, email)
- Number of promises needed

## Execution Protocol

### Step 1: Filter Eligible Candidates

```python
def filter_eligible_candidates(all_candidates):
    """
    Remove blocked and below-threshold candidates.
    """
    eligible = []
    removed = []

    for candidate in all_candidates:
        status = candidate['calibration_status']['status']

        if 'BLOCKED' in status or 'BELOW THRESHOLD' in status:
            removed.append({
                'id': candidate['promise_id'],
                'reason': status,
                'issues': candidate['calibration_status'].get('issues', [])
            })
        else:
            eligible.append(candidate)

    return eligible, removed
```

### Step 2: Calculate Final Ranking Score

```python
def calculate_ranking_score(candidate, thesis_data, all_candidates):
    """
    Calculate final ranking score for a candidate.
    """
    scores = {}

    # CALIBRATION COMPOSITE (50%)
    scores['calibration'] = candidate['calibration_composite']

    # THESIS ALIGNMENT (20%)
    thesis_promise = thesis_data.get('primary_thesis', {}).get('promise_used', '')
    candidate_promise = extract_core_promise(candidate['statement'])

    if candidate_promise == thesis_promise:
        scores['thesis_alignment'] = 10
    elif any(r.get('promise_used') == candidate_promise for r in thesis_data.get('runners_up', [])):
        scores['thesis_alignment'] = 8
    elif candidate['dimension_scores']['thesis_potential'] >= 7:
        scores['thesis_alignment'] = 6
    else:
        scores['thesis_alignment'] = 4

    # UNIQUENESS (15%)
    # How different is this from other top candidates?
    uniqueness_score = calculate_uniqueness(candidate, all_candidates)
    scores['uniqueness'] = uniqueness_score

    # VERSATILITY (15%)
    # Can this be used in multiple copy positions?
    versatility_score = assess_versatility(candidate)
    scores['versatility'] = versatility_score

    # WEIGHTED COMPOSITE
    weights = {
        'calibration': 0.50,
        'thesis_alignment': 0.20,
        'uniqueness': 0.15,
        'versatility': 0.15
    }

    final_score = sum(scores[dim] * weights[dim] for dim in weights)

    return {
        'final_score': round(final_score, 2),
        'component_scores': scores
    }


def calculate_uniqueness(candidate, all_candidates):
    """
    Score how unique this promise is compared to others.
    """
    candidate_core = extract_core_promise(candidate['statement'])

    similarity_scores = []
    for other in all_candidates:
        if other['promise_id'] != candidate['promise_id']:
            other_core = extract_core_promise(other['statement'])
            sim = calculate_similarity(candidate_core, other_core)
            similarity_scores.append(sim)

    # Lower similarity = higher uniqueness
    if not similarity_scores:
        return 10

    avg_similarity = sum(similarity_scores) / len(similarity_scores)
    uniqueness = 10 - (avg_similarity * 10)

    return max(1, min(10, uniqueness))


def assess_versatility(candidate):
    """
    Assess how versatile this promise is across copy positions.
    """
    versatility_factors = []

    # Length check (shorter = more versatile)
    word_count = len(candidate['statement'].split())
    if word_count <= 10:
        versatility_factors.append(3)
    elif word_count <= 15:
        versatility_factors.append(2)
    else:
        versatility_factors.append(1)

    # Mechanism-connected = more versatile (can strip or keep)
    if candidate.get('mechanism_connected'):
        versatility_factors.append(2)
    else:
        versatility_factors.append(1)

    # Clear promise type = more versatile
    if candidate.get('promise_type') in ['transformation', 'relief', 'improvement']:
        versatility_factors.append(2)
    else:
        versatility_factors.append(1)

    # Emotional universality
    frame = candidate.get('emotional_frame')
    if frame in ['hope', 'frustration_relief']:
        versatility_factors.append(2)
    elif frame:
        versatility_factors.append(1)
    else:
        versatility_factors.append(1.5)

    base_score = sum(versatility_factors)
    normalized = (base_score / 9) * 10  # Max possible = 9

    return min(10, round(normalized, 1))
```

### Step 3: Apply Secondary Considerations

```python
def apply_secondary_considerations(ranked_candidates):
    """
    Adjust rankings to ensure variety and coverage.
    """
    adjustments = []

    # Check promise type coverage in top 10
    top_10 = ranked_candidates[:10]
    types_in_top_10 = set(c.get('promise_type') for c in top_10 if c.get('promise_type'))

    if len(types_in_top_10) < 3:
        # Find candidates with underrepresented types
        all_types = ['transformation', 'improvement', 'relief', 'capability', 'prevention']
        missing_types = [t for t in all_types if t not in types_in_top_10]

        for candidate in ranked_candidates[10:20]:
            if candidate.get('promise_type') in missing_types:
                adjustments.append({
                    'id': candidate['promise_id'],
                    'adjustment': 'boost',
                    'reason': f"Type diversity: {candidate.get('promise_type')}"
                })
                candidate['final_score'] += 0.5  # Boost

    # Check emotional range in top 5
    top_5 = ranked_candidates[:5]
    frames_in_top_5 = set(c.get('emotional_frame') for c in top_5 if c.get('emotional_frame'))

    if len(frames_in_top_5) < 2:
        for candidate in ranked_candidates[5:15]:
            if candidate.get('emotional_frame') and candidate.get('emotional_frame') not in frames_in_top_5:
                adjustments.append({
                    'id': candidate['promise_id'],
                    'adjustment': 'boost',
                    'reason': f"Emotional diversity: {candidate.get('emotional_frame')}"
                })
                candidate['final_score'] += 0.3

    # Re-sort after adjustments
    ranked_candidates.sort(key=lambda x: x['final_score'], reverse=True)

    return ranked_candidates, adjustments
```

### Step 4: Generate Final Rankings

```python
def generate_final_rankings(eligible_candidates, thesis_data):
    """
    Produce final ranked list with annotations.
    """
    # Calculate ranking scores
    for candidate in eligible_candidates:
        ranking_result = calculate_ranking_score(candidate, thesis_data, eligible_candidates)
        candidate['final_score'] = ranking_result['final_score']
        candidate['ranking_components'] = ranking_result['component_scores']

    # Sort by final score
    ranked = sorted(eligible_candidates, key=lambda x: x['final_score'], reverse=True)

    # Apply secondary considerations
    ranked, adjustments = apply_secondary_considerations(ranked)

    # Assign ranks
    for i, candidate in enumerate(ranked):
        candidate['rank'] = i + 1
        candidate['tier'] = assign_final_tier(candidate, i)

    return {
        'ranked_candidates': ranked,
        'adjustments_applied': adjustments,
        'primary_promise': ranked[0] if ranked else None,
        'supporting_promises': ranked[1:5] if len(ranked) > 1 else [],
        'reserve_pool': ranked[5:15] if len(ranked) > 5 else []
    }


def assign_final_tier(candidate, position):
    """
    Assign final tier based on position and score.
    """
    if position == 0:
        return 'PRIMARY'
    elif position < 5:
        return 'SUPPORTING'
    elif position < 10:
        return 'SECONDARY'
    elif position < 15:
        return 'RESERVE'
    else:
        return 'ARCHIVE'
```

## Layer 2 Gate Criteria

```yaml
layer_2_gate:

  PASS:
    - primary_promise_selected: true
    - supporting_promises >= 3
    - all_supporting_promises_score >= 6.5
    - campaign_thesis_generated: true
    - promise_type_coverage >= 3

  PASS_WITH_WARNINGS:
    - PASS criteria met
    - BUT: reserve_pool < 5
    - OR: emotional_frame_coverage < 2

  FAIL:
    - no_primary_promise
    - OR supporting_promises < 2
    - OR no_campaign_thesis
    - OR all_promises_undifferentiated
```

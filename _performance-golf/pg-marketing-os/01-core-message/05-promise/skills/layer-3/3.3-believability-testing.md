# 3.3 Believability Testing

## Purpose
Test whether promises feel believable to the target audience—not just whether they're true or supported by proof. Believability is about perception: does the promise land as credible, or does it trigger the "yeah right" response?

---

## Input Requirements

From 3.2 Objection Resilience:
- Resilience-tested promises
- Objection mapping

From 01-research:
- Avatar data (what they've tried, what they believe)
- Market sophistication (Schwartz stage)
- Competitor promises (what's been said before)

From Context:
- Promise specificity level
- Mechanism familiarity

## Execution Protocol

### Step 1: Plausibility Assessment

```python
def assess_plausibility(promise, mechanism_data, market_data):
    """
    Assess whether the promise seems physically/logically possible.
    """
    score = 10  # Start at max, deduct for issues
    issues = []

    statement = promise['statement'].lower()

    # Check for impossible claims
    impossible_patterns = [
        ('overnight', -3),
        ('instant', -2),
        ('100%', -3),
        ('cure', -4),  # Can't say cure
        ('impossible', -2),
        ('miracle', -4)
    ]

    for pattern, penalty in impossible_patterns:
        if pattern in statement:
            score += penalty
            issues.append(f"Contains '{pattern}' - triggers disbelief")

    # Check mechanism backing
    if promise.get('mechanism_connected'):
        if mechanism_data.get('mechanism_type') == 'scientific_principle':
            score += 1  # Science backing helps
    else:
        score -= 2
        issues.append("No mechanism explains how this happens")

    # Check against known reality
    promise_magnitude = assess_promise_magnitude(promise)
    if promise_magnitude == 'extreme' and not promise.get('proof_verified'):
        score -= 3
        issues.append("Extreme claim without strong proof")

    return {
        'plausibility_score': max(1, min(10, score)),
        'issues': issues
    }


def assess_promise_magnitude(promise):
    """Determine if promise is modest, significant, or extreme."""
    statement = promise['statement'].lower()

    extreme_indicators = ['transform', 'completely', 'never again', 'cure', 'only way']
    significant_indicators = ['significantly', 'major', 'dramatic', 'finally']
    modest_indicators = ['improve', 'reduce', 'help', 'support', 'may']

    extreme_count = sum(1 for i in extreme_indicators if i in statement)
    significant_count = sum(1 for i in significant_indicators if i in statement)
    modest_count = sum(1 for i in modest_indicators if i in statement)

    if extreme_count >= 2:
        return 'extreme'
    elif significant_count >= 1 or extreme_count == 1:
        return 'significant'
    else:
        return 'modest'
```

### Step 2: Specificity Sweet Spot

```python
def assess_specificity_believability(promise):
    """
    Check if specificity level hits the believability sweet spot.
    """
    statement = promise['statement']

    # Extract specificity markers
    has_numbers = bool(re.search(r'\d+', statement))
    has_timeframe = bool(re.search(r'(\d+\s*(days?|weeks?|months?)|in\s+just)', statement.lower()))
    has_qualifier = any(q in statement.lower() for q in ['up to', 'as little as', 'as much as', 'many', 'some'])

    # Assess each
    issues = []
    bonuses = []

    if has_numbers:
        # Check if numbers are believable
        numbers = re.findall(r'\d+', statement)
        for num in numbers:
            n = int(num)
            if n in [7, 14, 21, 30]:  # Common timeframes
                bonuses.append(f"{n} is a familiar timeframe")
            elif n == 47 or n == 23 or n == 67:  # Specific = credible
                bonuses.append(f"{n} is specific (not round)")
            elif n == 100 or n == 50 or n == 10:
                issues.append(f"{n} is too round—feels made up")
    else:
        issues.append("No specific numbers—harder to believe")

    if has_qualifier:
        bonuses.append("Qualifier present—doesn't overclaim")
    else:
        if has_numbers:
            issues.append("Specific number without qualifier may overpromise")

    # Score
    score = 5 + len(bonuses) - len(issues)

    return {
        'specificity_believability': max(1, min(10, score)),
        'has_numbers': has_numbers,
        'has_qualifier': has_qualifier,
        'has_timeframe': has_timeframe,
        'bonuses': bonuses,
        'issues': issues
    }
```

### Step 3: Social Validation Check

```python
def assess_social_validation(promise, proof_mapping):
    """
    Check if testimonials/social proof supports believability.
    """
    # Find social proof for this promise
    social_proof = [
        p for p in proof_mapping.get('proof_matches', [])
        if p['proof_type'] == 'social'
    ]

    if not social_proof:
        return {
            'social_validation_score': 4,
            'issues': ['No testimonials support this specific claim'],
            'recommendation': 'Need testimonials that echo this promise'
        }

    # Check testimonial variety
    score = 5
    bonuses = []

    if len(social_proof) >= 3:
        score += 2
        bonuses.append("Multiple testimonials support claim")

    # Check for relatable testimonials
    has_specific_result = any('specific_result' in str(p) for p in social_proof)
    has_transformation = any('transformation' in str(p) for p in social_proof)

    if has_specific_result:
        score += 1
        bonuses.append("Testimonials include specific results")

    if has_transformation:
        score += 1
        bonuses.append("Transformation story validates journey")

    return {
        'social_validation_score': min(10, score),
        'testimonial_count': len(social_proof),
        'bonuses': bonuses
    }
```

### Step 4: Effort-Outcome Alignment

```python
def assess_effort_alignment(promise):
    """
    Check if effort implied matches outcome promised.
    """
    statement = promise['statement'].lower()

    # Detect effort level implied
    low_effort_markers = ['without', 'effortless', 'easy', 'simple', 'just', 'automatically']
    high_effort_markers = ['work', 'commit', 'dedicate', 'follow', 'protocol', 'system']

    implied_effort = 'moderate'
    if any(m in statement for m in low_effort_markers):
        implied_effort = 'low'
    elif any(m in statement for m in high_effort_markers):
        implied_effort = 'high'

    # Detect outcome magnitude
    magnitude = assess_promise_magnitude(promise)

    # Check alignment
    alignment_matrix = {
        ('low', 'extreme'): {'score': 3, 'issue': 'Big promise with no effort—skepticism trigger'},
        ('low', 'significant'): {'score': 6, 'issue': 'Needs mechanism explanation for low effort'},
        ('low', 'modest'): {'score': 9, 'issue': None},
        ('moderate', 'extreme'): {'score': 5, 'issue': 'Moderate effort may not justify extreme result'},
        ('moderate', 'significant'): {'score': 8, 'issue': None},
        ('moderate', 'modest'): {'score': 9, 'issue': None},
        ('high', 'extreme'): {'score': 8, 'issue': None},
        ('high', 'significant'): {'score': 9, 'issue': None},
        ('high', 'modest'): {'score': 7, 'issue': 'High effort for modest result may deter'}
    }

    result = alignment_matrix.get((implied_effort, magnitude), {'score': 6, 'issue': None})

    return {
        'effort_alignment_score': result['score'],
        'implied_effort': implied_effort,
        'outcome_magnitude': magnitude,
        'issue': result['issue'],
        'recommendation': 'Add mechanism explanation' if result['score'] < 6 else None
    }
```

### Step 5: Calculate Overall Believability

```python
def calculate_believability(promise, assessments):
    """
    Calculate composite believability score.
    """
    weights = {
        'plausibility': 0.30,
        'specificity_believability': 0.25,
        'social_validation': 0.20,
        'effort_alignment': 0.15,
        'novelty': 0.10
    }

    scores = {
        'plausibility': assessments['plausibility']['plausibility_score'],
        'specificity_believability': assessments['specificity']['specificity_believability'],
        'social_validation': assessments['social']['social_validation_score'],
        'effort_alignment': assessments['effort']['effort_alignment_score'],
        'novelty': assessments.get('novelty', {}).get('novelty_score', 7)
    }

    composite = sum(scores[dim] * weights[dim] for dim in weights)

    # Categorize
    if composite >= 8:
        status = 'HIGHLY_BELIEVABLE'
    elif composite >= 6.5:
        status = 'BELIEVABLE'
    elif composite >= 5:
        status = 'MARGINAL'
    else:
        status = 'LOW_BELIEVABILITY'

    return {
        'believability_score': round(composite, 2),
        'component_scores': scores,
        'believability_status': status,
        'all_issues': collect_all_issues(assessments),
        'recommendations': generate_believability_fixes(assessments, status)
    }
```

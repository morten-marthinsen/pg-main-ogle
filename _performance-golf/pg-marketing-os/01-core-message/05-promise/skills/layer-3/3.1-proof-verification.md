# 3.1 Proof Verification

## Purpose
Verify that specific proof exists for each ranked promise. Not general proof—specific proof that backs the exact claim being made. A promise without verified proof is a liability.

---

## Input Requirements

From 2.7 Top Candidate Ranking:
- Ranked promise candidates
- Primary promise
- Supporting promises

From 02-proof-inventory:
- Complete proof inventory
- Proof element scores
- Proof categories and sub-types

## Execution Protocol

### Step 1: Match Promises to Proof Elements

```python
def match_promise_to_proof(promise, proof_inventory):
    """
    Find proof elements that support this specific promise.
    """
    matches = []
    promise_core = extract_core_promise(promise['statement'])
    promise_keywords = extract_claim_keywords(promise_core)

    for proof in proof_inventory:
        # Calculate match strength
        match_score = 0
        match_reasons = []

        # Direct keyword match
        proof_keywords = extract_proof_keywords(proof)
        keyword_overlap = len(set(promise_keywords) & set(proof_keywords))
        if keyword_overlap >= 2:
            match_score += 3
            match_reasons.append(f"Keyword match: {keyword_overlap} keywords")

        # Semantic similarity
        semantic_sim = calculate_semantic_similarity(promise_core, proof.get('claim_supported', ''))
        if semantic_sim > 0.7:
            match_score += 3
            match_reasons.append(f"Semantic match: {semantic_sim:.2f}")
        elif semantic_sim > 0.5:
            match_score += 1
            match_reasons.append(f"Partial semantic match: {semantic_sim:.2f}")

        # Promise type alignment
        if promise_type_matches_proof_type(promise.get('promise_type'), proof.get('category')):
            match_score += 2
            match_reasons.append("Type alignment")

        # If any match found
        if match_score >= 2:
            matches.append({
                'proof_id': proof['id'],
                'proof_summary': proof.get('summary'),
                'proof_type': proof.get('category'),
                'proof_score': proof.get('score', 5),
                'match_score': match_score,
                'match_reasons': match_reasons,
                'direct_match': keyword_overlap >= 3 or semantic_sim > 0.8
            })

    # Sort by match score
    matches.sort(key=lambda x: (x['direct_match'], x['match_score'], x['proof_score']), reverse=True)

    return matches


def promise_type_matches_proof_type(promise_type, proof_category):
    """
    Check if proof category naturally supports promise type.
    """
    type_to_category = {
        'transformation': ['social', 'demonstration'],
        'improvement': ['data', 'demonstration'],
        'relief': ['social', 'mechanism'],
        'capability': ['demonstration', 'authority'],
        'prevention': ['data', 'authority']
    }

    preferred_categories = type_to_category.get(promise_type, [])
    return proof_category in preferred_categories
```

### Step 2: Assess Verification Level

```python
def assess_verification_level(promise, proof_matches):
    """
    Determine verification level based on proof matches.
    """
    if not proof_matches:
        return {
            'level': 'unverified',
            'strength': 0,
            'issues': ['No matching proof found'],
            'action': 'Must downgrade, reframe, or remove'
        }

    # Best match analysis
    best_match = proof_matches[0]
    direct_matches = [m for m in proof_matches if m['direct_match']]
    total_proof_strength = sum(m['proof_score'] for m in proof_matches[:3])

    # Knockout level
    if best_match['direct_match'] and best_match['proof_score'] >= 8:
        return {
            'level': 'verified_knockout',
            'strength': 10,
            'primary_proof': best_match,
            'supporting_proof': proof_matches[1:3],
            'issues': []
        }

    # Strong level
    if len(direct_matches) >= 2 or (best_match['direct_match'] and total_proof_strength >= 15):
        return {
            'level': 'verified_strong',
            'strength': 8,
            'primary_proof': best_match,
            'supporting_proof': proof_matches[1:3],
            'issues': []
        }

    # Adequate level
    if best_match['proof_score'] >= 6:
        return {
            'level': 'verified_adequate',
            'strength': 6,
            'primary_proof': best_match,
            'supporting_proof': proof_matches[1:2] if len(proof_matches) > 1 else [],
            'issues': ['May need additional proof in copy'],
            'copy_guidance': 'Stack multiple proof elements when presenting this promise'
        }

    # Weak level
    if proof_matches:
        return {
            'level': 'verified_weak',
            'strength': 4,
            'primary_proof': best_match,
            'issues': ['Proof is indirect or thin', 'Consider softening claim language'],
            'copy_guidance': 'Use hedging language or present as secondary benefit'
        }

    return {
        'level': 'unverified',
        'strength': 0,
        'issues': ['Proof insufficient for claim'],
        'action': 'Downgrade claim'
    }
```

### Step 3: Generate Proof-Promise Mapping

```python
def generate_proof_mapping(ranked_promises, proof_inventory):
    """
    Create comprehensive proof mapping for all ranked promises.
    """
    mapping = []

    for promise in ranked_promises:
        # Find matching proof
        matches = match_promise_to_proof(promise, proof_inventory)

        # Assess verification
        verification = assess_verification_level(promise, matches)

        mapping.append({
            'promise_id': promise['promise_id'],
            'promise': promise['statement'],
            'rank': promise['rank'],
            'tier': promise['tier'],
            'proof_matches': matches[:5],
            'verification': verification,
            'proof_integration_guidance': generate_proof_guidance(promise, verification)
        })

    return mapping


def generate_proof_guidance(promise, verification):
    """
    Generate specific guidance for using proof with this promise.
    """
    level = verification['level']
    primary = verification.get('primary_proof')

    if level == 'verified_knockout':
        return {
            'approach': 'LEAD WITH PROOF',
            'guidance': f"Can lead with {primary['proof_type']} proof when presenting this promise",
            'example': f"Use {primary['proof_summary'][:50]}... as immediate validation"
        }

    elif level == 'verified_strong':
        return {
            'approach': 'PROOF STACK',
            'guidance': "Stack multiple proof elements immediately after promise",
            'example': "Promise → Primary proof → Supporting proof → Expansion"
        }

    elif level == 'verified_adequate':
        return {
            'approach': 'CLAIM THEN PROVE',
            'guidance': "Make claim, then provide proof, but don't linger—move to next point",
            'example': "State promise → Single proof element → Transition to mechanism"
        }

    elif level == 'verified_weak':
        return {
            'approach': 'SOFT CLAIM',
            'guidance': "Use softening language and embed in larger proof sequence",
            'example': "Present as 'many users report...' rather than direct claim"
        }

    else:
        return {
            'approach': 'DO NOT CLAIM',
            'guidance': "Cannot claim this without additional proof",
            'action': 'Reframe as aspiration or remove entirely'
        }
```

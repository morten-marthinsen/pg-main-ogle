# 3.6 Mechanism Story Coherence

## Purpose
Verify that each promise fits coherently within the mechanism narrative arc. The mechanism tells a story—root cause → villain → solution → transformation. Promises must fit logically within this story without contradicting or jumping ahead.

---

## Input Requirements

From 3.5 Customer Voice Validation:
- Voice-validated promises

From 04-mechanism:
- Mechanism name
- Mechanism story arc
- Villain
- Villain defeat action
- Expected outcomes

From 03-root-cause:
- Root cause identified
- Problem chain
- Countersell targets

---


### Step 2: Check Logical Sequence

```python
def check_logical_sequence(promise, story_arc):
    """
    Verify promise follows logically from mechanism story.
    """
    issues = []
    score = 10

    statement = promise['statement'].lower()
    mechanism_name = story_arc['mechanism']['name'].lower()
    villain = story_arc['root_cause'].get('villain', '').lower()
    expected_outcomes = [o.lower() for o in story_arc['transformation']['expected_outcomes']]

    # Check 1: If promise mentions mechanism, does it connect correctly?
    if mechanism_name in statement:
        how_it_works = story_arc['mechanism'].get('how_it_works', '').lower()
        # Promise should align with mechanism's stated function
        promise_core = extract_core_promise(promise['statement'])

        if not any(outcome in promise_core for outcome in expected_outcomes):
            if not check_semantic_alignment(promise_core, expected_outcomes):
                score -= 2
                issues.append("Promise outcome not in mechanism's expected outcomes")

    # Check 2: If promise mentions villain, is defeat action implied?
    if villain and villain in statement:
        defeat_action = story_arc['villain_defeat'].get('defeat_action', '').lower()
        if not any(word in statement for word in defeat_action.split()[:3]):
            score -= 1
            issues.append("Villain mentioned but defeat action unclear")

    # Check 3: Does promise skip steps in the arc?
    if 'transform' in statement:
        # Transformation should come after mechanism
        if not promise.get('mechanism_connected'):
            score -= 2
            issues.append("Transformation promised without mechanism connection")

    return {
        'logical_sequence_score': max(1, score),
        'issues': issues,
        'flows_logically': len(issues) == 0
    }
```

### Step 3: Check for Contradictions

```python
def check_contradictions(promise, story_arc, all_promises):
    """
    Check if promise contradicts mechanism claims or other promises.
    """
    contradictions = []
    score = 10

    statement = promise['statement'].lower()

    # Check against mechanism claims
    mechanism_claims = story_arc['mechanism'].get('how_it_works', '')
    root_cause = story_arc['root_cause'].get('real_cause', '')

    # Contradiction patterns
    contradiction_checks = [
        # If mechanism is about gut, promise shouldn't claim brain-based solution
        {
            'mechanism_pattern': 'gut',
            'contradicts': ['brain-based', 'neural', 'psychological'],
            'unless': ['gut-brain', 'gut brain', 'connection']
        },
        # If mechanism is slow/sustained, promise shouldn't claim instant
        {
            'mechanism_pattern': 'protocol',
            'contradicts': ['instant', 'immediate', 'overnight'],
            'unless': ['start seeing', 'begin to']
        }
    ]

    for check in contradiction_checks:
        if check['mechanism_pattern'] in mechanism_claims.lower():
            for contra in check['contradicts']:
                if contra in statement and not any(exc in statement for exc in check['unless']):
                    contradictions.append(f"Promise says '{contra}' but mechanism is '{check['mechanism_pattern']}'-based")
                    score -= 3

    # Check against other promises in the set (for consistency)
    for other in all_promises:
        if other['promise_id'] != promise['promise_id']:
            if check_promise_conflict(promise, other):
                contradictions.append(f"Potential conflict with {other['promise_id']}")
                score -= 1

    return {
        'contradiction_score': max(1, score),
        'contradictions': contradictions,
        'no_contradictions': len(contradictions) == 0
    }


def check_promise_conflict(promise1, promise2):
    """Check if two promises conflict with each other."""
    # Example: One promises "effortless" while another implies "hard work"
    effort_conflict = (
        ('effortless' in promise1['statement'].lower() and 'work' in promise2['statement'].lower()) or
        ('effortless' in promise2['statement'].lower() and 'work' in promise1['statement'].lower())
    )

    # Time conflict
    time_conflict = (
        ('fast' in promise1['statement'].lower() and 'time' in promise2['statement'].lower()) or
        ('instant' in promise1['statement'].lower() and 'gradually' in promise2['statement'].lower())
    )

    return effort_conflict or time_conflict
```

### Step 4: Check Villain Connection

```python
def check_villain_connection(promise, story_arc):
    """
    Verify promise connects to villain defeat narrative.
    """
    villain = story_arc['root_cause'].get('villain')
    defeat_action = story_arc['villain_defeat'].get('defeat_action')
    defeat_result = story_arc['villain_defeat'].get('defeat_result')

    if not villain:
        return {
            'villain_connection_score': 7,
            'has_villain_connection': False,
            'note': 'No villain in mechanism—neutral'
        }

    statement = promise['statement'].lower()
    villain_lower = villain.lower()

    # Direct villain mention
    if villain_lower in statement:
        return {
            'villain_connection_score': 10,
            'has_villain_connection': True,
            'connection_type': 'direct_mention'
        }

    # Defeat result connection
    if defeat_result and any(word in statement for word in defeat_result.lower().split()[:3]):
        return {
            'villain_connection_score': 8,
            'has_villain_connection': True,
            'connection_type': 'defeat_result_implied'
        }

    # Could connect but doesn't
    promise_core = extract_core_promise(promise['statement'])
    villain_problems = story_arc['root_cause'].get('villain_actions', [])

    if any(p.lower() in promise_core.lower() for p in villain_problems):
        return {
            'villain_connection_score': 6,
            'has_villain_connection': False,
            'connection_type': 'implicit_only',
            'recommendation': f"Could strengthen by explicitly linking to '{villain}'"
        }

    return {
        'villain_connection_score': 5,
        'has_villain_connection': False,
        'connection_type': 'none',
        'note': 'Promise doesn\'t connect to villain narrative'
    }
```

### Step 5: Check Scope Alignment

```python
def check_scope_alignment(promise, story_arc):
    """
    Verify promise scope matches mechanism scope.
    """
    issues = []
    score = 10

    mechanism_scope = infer_mechanism_scope(story_arc['mechanism'])
    promise_scope = infer_promise_scope(promise)

    # Scope levels: narrow, moderate, broad, unlimited
    scope_compatibility = {
        ('narrow', 'narrow'): 10,
        ('narrow', 'moderate'): 7,
        ('narrow', 'broad'): 4,
        ('narrow', 'unlimited'): 2,
        ('moderate', 'narrow'): 9,
        ('moderate', 'moderate'): 10,
        ('moderate', 'broad'): 6,
        ('moderate', 'unlimited'): 3,
        ('broad', 'narrow'): 9,
        ('broad', 'moderate'): 9,
        ('broad', 'broad'): 10,
        ('broad', 'unlimited'): 5
    }

    score = scope_compatibility.get((mechanism_scope, promise_scope), 6)

    if score < 6:
        issues.append(f"Promise scope ({promise_scope}) exceeds mechanism scope ({mechanism_scope})")

    return {
        'scope_alignment_score': score,
        'mechanism_scope': mechanism_scope,
        'promise_scope': promise_scope,
        'scope_aligned': score >= 6,
        'issues': issues
    }


def infer_mechanism_scope(mechanism):
    """Determine scope of mechanism claims."""
    mech_type = mechanism.get('type', '')
    outcomes = len(mechanism.get('expected_outcomes', []))

    if outcomes <= 2:
        return 'narrow'
    elif outcomes <= 5:
        return 'moderate'
    else:
        return 'broad'


def infer_promise_scope(promise):
    """Determine scope of promise claim."""
    statement = promise['statement'].lower()

    unlimited_markers = ['everything', 'all aspects', 'entire life', 'completely transform']
    broad_markers = ['multiple', 'many', 'various', 'transform']
    narrow_markers = ['specifically', 'one thing', 'this one']

    if any(m in statement for m in unlimited_markers):
        return 'unlimited'
    elif any(m in statement for m in broad_markers):
        return 'broad'
    elif any(m in statement for m in narrow_markers):
        return 'narrow'
    else:
        return 'moderate'
```

### Step 6: Calculate Coherence Score

```python
def calculate_coherence_score(promise, checks):
    """
    Calculate overall story coherence score.
    """
    weights = {
        'logical_sequence': 0.30,
        'no_contradictions': 0.30,
        'villain_connection': 0.20,
        'scope_alignment': 0.20
    }

    scores = {
        'logical_sequence': checks['sequence']['logical_sequence_score'],
        'no_contradictions': checks['contradictions']['contradiction_score'],
        'villain_connection': checks['villain']['villain_connection_score'],
        'scope_alignment': checks['scope']['scope_alignment_score']
    }

    composite = sum(scores[dim] * weights[dim] for dim in weights)

    # Status
    if composite >= 8:
        status = 'FULLY_COHERENT'
    elif composite >= 6.5:
        status = 'COHERENT'
    elif composite >= 5:
        status = 'MINOR_ISSUES'
    else:
        status = 'INCOHERENT'

    return {
        'coherence_score': round(composite, 2),
        'component_scores': scores,
        'coherence_status': status,
        'issues': collect_all_coherence_issues(checks)
    }
```

# 3.5 Customer Voice Validation

## Purpose
Validate that promises sound like how customers actually talk about their problems and desires. Promises that use marketer language instead of customer language fail to resonate. This microskill ensures promises speak the prospect's language.

---

## Input Requirements

From 3.4 Compliance Check:
- Compliance-checked promises

From 01-research:
- Customer language library
- Verbatim quotes from testimonials/reviews
- Common phrases and expressions
- Avatar data

From 1.6 Customer Language Mapping:
- Customer language scores from Layer 1

## Execution Protocol

### Step 1: Build Customer Language Baseline

```python
def build_language_baseline(research_data):
    """
    Extract customer language patterns from research.
    """
    baseline = {
        'problem_vocabulary': [],
        'desire_vocabulary': [],
        'emotion_words': [],
        'common_phrases': [],
        'numbers_used': [],
        'timeframes_mentioned': [],
        'failed_solutions': [],
        'marketer_words_to_avoid': []
    }

    # From testimonials
    for testimonial in research_data.get('testimonials', []):
        text = testimonial.get('quote', '')

        # Extract problem language
        problem_phrases = extract_problem_phrases(text)
        baseline['problem_vocabulary'].extend(problem_phrases)

        # Extract desire language
        desire_phrases = extract_desire_phrases(text)
        baseline['desire_vocabulary'].extend(desire_phrases)

        # Extract numbers
        numbers = extract_numbers_mentioned(text)
        baseline['numbers_used'].extend(numbers)

    # From reviews
    for review in research_data.get('reviews', []):
        text = review.get('text', '')
        baseline['common_phrases'].extend(extract_common_phrases(text))

    # Marketer words to avoid (not found in customer language)
    baseline['marketer_words_to_avoid'] = [
        'leverage', 'optimize', 'synergy', 'paradigm', 'revolutionary',
        'breakthrough', 'proprietary', 'cutting-edge', 'state-of-the-art',
        'best-in-class', 'world-class', 'game-changer'
    ]

    # Deduplicate and rank by frequency
    for key in baseline:
        if isinstance(baseline[key], list):
            baseline[key] = rank_by_frequency(baseline[key])

    return baseline


def extract_problem_phrases(text):
    """Extract how customers describe their problems."""
    problem_patterns = [
        r"I('ve| have) been struggling with",
        r"I can't seem to",
        r"No matter what I try",
        r"I('m| am) so tired of",
        r"I hate when",
        r"It's so frustrating",
        r"I've tried everything"
    ]

    phrases = []
    text_lower = text.lower()

    for pattern in problem_patterns:
        matches = re.findall(pattern + r'[^.!?]+', text_lower)
        phrases.extend(matches)

    return phrases


def extract_desire_phrases(text):
    """Extract how customers express desires."""
    desire_patterns = [
        r"I (just )?want(ed)? to",
        r"I wish I could",
        r"Finally I can",
        r"Now I('m| am) able to",
        r"I dream(ed)? of",
        r"All I want is"
    ]

    phrases = []
    text_lower = text.lower()

    for pattern in desire_patterns:
        matches = re.findall(pattern + r'[^.!?]+', text_lower)
        phrases.extend(matches)

    return phrases
```

### Step 2: Validate Vocabulary Match

```python
def validate_vocabulary(promise, baseline):
    """
    Check if promise uses customer vocabulary.
    """
    statement = promise['statement'].lower()
    words = statement.split()

    # Check for marketer jargon
    jargon_found = []
    for word in words:
        if word in baseline['marketer_words_to_avoid']:
            jargon_found.append(word)

    # Check for customer vocabulary overlap
    customer_words = set()
    for phrase_list in [baseline['problem_vocabulary'], baseline['desire_vocabulary']]:
        for phrase in phrase_list:
            customer_words.update(phrase.lower().split())

    overlap = set(words) & customer_words
    overlap_ratio = len(overlap) / len(words) if words else 0

    # Score
    score = 7  # Base score

    if jargon_found:
        score -= len(jargon_found) * 1.5
    else:
        score += 1  # Bonus for no jargon

    if overlap_ratio >= 0.4:
        score += 2
    elif overlap_ratio >= 0.2:
        score += 1
    else:
        score -= 1

    return {
        'vocabulary_score': max(1, min(10, score)),
        'jargon_found': jargon_found,
        'customer_overlap': list(overlap),
        'overlap_ratio': round(overlap_ratio, 2),
        'recommendation': 'Remove jargon' if jargon_found else None
    }
```

### Step 3: Validate Emotional Resonance

```python
def validate_emotional_resonance(promise, baseline, customer_emotions):
    """
    Check if emotional framing matches customer expressions.
    """
    statement = promise['statement'].lower()

    # Map promise emotion to customer emotion language
    promise_emotion = promise.get('emotional_frame')

    emotion_word_mapping = {
        'hope': ['finally', 'at last', 'dream', 'wish', 'someday'],
        'frustration_relief': ['tired of', 'sick of', 'fed up', 'no more', 'end'],
        'fear_escape': ['scared', 'worried', 'afraid', 'never have to'],
        'vindication': ['prove', 'show them', 'was right', 'all along'],
        'freedom': ['free', 'finally free', 'no longer', 'without'],
        'control': ['in control', 'my choice', 'decide', 'power']
    }

    # Check for emotion alignment
    expected_words = emotion_word_mapping.get(promise_emotion, [])
    found_emotion_words = [w for w in expected_words if w in statement]

    # Check against customer emotion expressions
    customer_emotion_phrases = customer_emotions.get(promise_emotion, [])
    matches_customer = any(phrase in statement for phrase in customer_emotion_phrases)

    # Score
    score = 5

    if found_emotion_words:
        score += len(found_emotion_words)

    if matches_customer:
        score += 2

    # Check intensity calibration
    intensity_match = check_intensity_match(promise, baseline)
    if intensity_match:
        score += 1

    return {
        'emotional_resonance_score': min(10, score),
        'emotion_words_found': found_emotion_words,
        'matches_customer_expression': matches_customer,
        'intensity_calibrated': intensity_match
    }
```

### Step 4: Validate Problem Framing

```python
def validate_problem_framing(promise, baseline):
    """
    Check if the promise frames the problem the way customers do.
    """
    statement = promise['statement'].lower()

    # Check for customer problem phrases
    problem_matches = []
    for phrase in baseline['problem_vocabulary'][:20]:  # Top 20
        phrase_lower = phrase.lower()
        # Check for partial matches
        phrase_words = set(phrase_lower.split())
        statement_words = set(statement.split())

        if len(phrase_words & statement_words) >= 2:
            problem_matches.append(phrase)

    # Check for failed solution acknowledgment
    failed_solution_patterns = [
        'tried everything', 'nothing worked', 'didn\'t work',
        'years of', 'after trying', 'finally'
    ]
    acknowledges_failed = any(p in statement for p in failed_solution_patterns)

    # Score
    score = 5

    if problem_matches:
        score += min(3, len(problem_matches))

    if acknowledges_failed:
        score += 2

    return {
        'problem_framing_score': min(10, score),
        'problem_phrase_matches': problem_matches,
        'acknowledges_failed_solutions': acknowledges_failed,
        'recommendation': 'Add failed solution acknowledgment' if not acknowledges_failed and promise.get('promise_type') == 'relief' else None
    }
```

### Step 5: Calculate Overall Voice Score

```python
def calculate_voice_score(promise, validations):
    """
    Calculate composite customer voice score.
    """
    weights = {
        'vocabulary_match': 0.30,
        'emotional_resonance': 0.25,
        'problem_framing': 0.25,
        'specificity_match': 0.20
    }

    scores = {
        'vocabulary_match': validations['vocabulary']['vocabulary_score'],
        'emotional_resonance': validations['emotion']['emotional_resonance_score'],
        'problem_framing': validations['problem']['problem_framing_score'],
        'specificity_match': validations.get('specificity', {}).get('specificity_score', 7)
    }

    composite = sum(scores[dim] * weights[dim] for dim in weights)

    # Determine status
    if composite >= 8:
        status = 'STRONG_CUSTOMER_VOICE'
    elif composite >= 6.5:
        status = 'GOOD_CUSTOMER_VOICE'
    elif composite >= 5:
        status = 'ADEQUATE'
    else:
        status = 'NEEDS_REWORK'

    return {
        'voice_score': round(composite, 2),
        'component_scores': scores,
        'voice_status': status,
        'improvements': collect_voice_improvements(validations)
    }


def collect_voice_improvements(validations):
    """Collect all recommended voice improvements."""
    improvements = []

    if validations['vocabulary'].get('jargon_found'):
        improvements.append({
            'type': 'remove_jargon',
            'detail': f"Remove: {', '.join(validations['vocabulary']['jargon_found'])}"
        })

    if validations['vocabulary']['overlap_ratio'] < 0.2:
        improvements.append({
            'type': 'add_customer_words',
            'detail': 'Incorporate more customer vocabulary'
        })

    if validations['emotion'].get('emotional_resonance_score', 0) < 6:
        improvements.append({
            'type': 'strengthen_emotion',
            'detail': 'Add emotional trigger words customers use'
        })

    if validations['problem'].get('recommendation'):
        improvements.append({
            'type': 'problem_framing',
            'detail': validations['problem']['recommendation']
        })

    return improvements
```

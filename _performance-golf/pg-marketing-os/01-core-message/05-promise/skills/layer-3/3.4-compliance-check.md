# 3.4 Compliance Check

## Purpose
Screen promises for FTC, legal, and platform compliance issues. A great promise that gets your ad account banned or triggers legal action is worthless. This microskill catches compliance red flags before copy is written.

---

## Input Requirements

From 3.3 Believability Testing:
- Believability-tested promises
- Issue flags from previous testing

From Context:
- Product category (supplement, info product, service, etc.)
- Target platforms (Facebook, Google, email)
- Jurisdiction (US, UK, etc.)

---

## Execution Protocol

### Step 1: Categorize Promise by Risk Level

```python
def categorize_compliance_risk(promise, product_category):
    """
    Determine initial compliance risk level.
    """
    statement = promise['statement'].lower()

    risk_level = 'low'
    risk_factors = []

    # High-risk categories
    high_risk_categories = ['supplement', 'health', 'weight_loss', 'income', 'financial']
    if product_category in high_risk_categories:
        risk_level = 'elevated'
        risk_factors.append(f"Product category: {product_category}")

    # High-risk language patterns
    high_risk_patterns = {
        'cure': 'health_claim',
        'treat': 'health_claim',
        'diagnose': 'health_claim',
        'prevent': 'health_claim',
        'guaranteed': 'guarantee_claim',
        'make money': 'income_claim',
        'earn': 'income_claim',
        'income': 'income_claim',
        'results may vary': 'needs_disclaimer'
    }

    for pattern, claim_type in high_risk_patterns.items():
        if pattern in statement:
            risk_level = 'high'
            risk_factors.append(f"Contains '{pattern}' - {claim_type}")

    # Moderate risk patterns
    moderate_risk_patterns = {
        'lose': 'weight_claim',
        'pounds': 'weight_claim',
        'fat': 'weight_claim',
        'always': 'absolute_claim',
        'never': 'absolute_claim',
        '100%': 'absolute_claim',
        'only': 'exclusivity_claim'
    }

    for pattern, claim_type in moderate_risk_patterns.items():
        if pattern in statement:
            if risk_level != 'high':
                risk_level = 'moderate'
            risk_factors.append(f"Contains '{pattern}' - {claim_type}")

    return {
        'risk_level': risk_level,
        'risk_factors': risk_factors,
        'requires_review': risk_level in ['moderate', 'high'],
        'requires_legal': risk_level == 'high'
    }
```

### Step 2: FTC Compliance Check

```python
def check_ftc_compliance(promise, proof_mapping):
    """
    Check promise against FTC advertising guidelines.
    """
    issues = []
    warnings = []
    statement = promise['statement']

    # Substantiation check
    has_substantiation = proof_mapping.get('verification', {}).get('level') in [
        'verified_knockout', 'verified_strong', 'verified_adequate'
    ]

    if not has_substantiation:
        issues.append({
            'issue': 'SUBSTANTIATION_REQUIRED',
            'description': 'FTC requires "competent and reliable scientific evidence" for claims',
            'action': 'Must have proof that substantiates this specific claim'
        })

    # Typical results check
    contains_result = any(w in statement.lower() for w in ['lose', 'gain', 'earn', 'achieve', 'get'])
    if contains_result:
        warnings.append({
            'warning': 'TYPICAL_RESULTS',
            'description': 'If results are atypical, must disclose typical results',
            'action': 'Add "Results not typical" or disclose typical results'
        })

    # Guarantee language check
    if 'guarantee' in statement.lower():
        issues.append({
            'issue': 'GUARANTEE_LANGUAGE',
            'description': 'Guarantees must be clearly explained and honored',
            'action': 'Ensure guarantee terms are specific and deliverable'
        })

    # Testimonial check
    if promise.get('uses_testimonial'):
        warnings.append({
            'warning': 'TESTIMONIAL_DISCLOSURE',
            'description': 'Testimonials require disclosure if compensated or atypical',
            'action': 'Include disclosure for compensated testimonials'
        })

    compliance_score = 10 - (len(issues) * 3) - (len(warnings) * 1)

    return {
        'ftc_compliance_score': max(0, compliance_score),
        'issues': issues,
        'warnings': warnings,
        'compliant': len(issues) == 0
    }
```

### Step 3: Health/Supplement Compliance

```python
def check_health_compliance(promise, product_category):
    """
    Check health-related claims for FDA compliance.
    """
    if product_category not in ['supplement', 'health', 'wellness', 'weight_loss']:
        return {'applicable': False}

    issues = []
    warnings = []
    required_disclaimers = []
    statement = promise['statement'].lower()

    # Disease claim check (prohibited for supplements)
    disease_terms = [
        'cure', 'treat', 'prevent', 'diagnose', 'disease',
        'diabetes', 'cancer', 'heart disease', 'alzheimer',
        'arthritis', 'depression', 'anxiety'
    ]

    for term in disease_terms:
        if term in statement:
            issues.append({
                'issue': 'DISEASE_CLAIM',
                'description': f"Cannot claim supplement treats/cures '{term}'",
                'severity': 'HIGH',
                'action': 'Remove disease language; use structure/function claims only'
            })

    # Structure/function language check
    acceptable_terms = [
        'supports', 'promotes', 'helps maintain', 'may help',
        'assists', 'contributes to'
    ]
    has_acceptable_language = any(term in statement for term in acceptable_terms)

    if product_category == 'supplement' and not has_acceptable_language:
        warnings.append({
            'warning': 'STRUCTURE_FUNCTION',
            'description': 'Consider using structure/function language',
            'example': '"Supports healthy gut function" instead of "Fixes your gut"'
        })

    # Required disclaimers
    if product_category == 'supplement':
        required_disclaimers.append({
            'disclaimer': 'FDA_DISCLAIMER',
            'text': '*These statements have not been evaluated by the FDA. This product is not intended to diagnose, treat, cure, or prevent any disease.',
            'placement': 'Required near claims'
        })

    compliance_score = 10 - (len(issues) * 4) - (len(warnings) * 1)

    return {
        'health_compliance_score': max(0, compliance_score),
        'applicable': True,
        'issues': issues,
        'warnings': warnings,
        'required_disclaimers': required_disclaimers,
        'compliant': len(issues) == 0
    }
```

### Step 4: Platform Policy Check

```python
def check_platform_compliance(promise, target_platforms):
    """
    Check against specific platform policies.
    """
    results = {}

    for platform in target_platforms:
        platform_issues = []
        platform_warnings = []

        if platform == 'facebook':
            platform_issues, platform_warnings = check_facebook_compliance(promise)
        elif platform == 'google':
            platform_issues, platform_warnings = check_google_compliance(promise)
        elif platform == 'youtube':
            platform_issues, platform_warnings = check_youtube_compliance(promise)

        results[platform] = {
            'issues': platform_issues,
            'warnings': platform_warnings,
            'compliant': len(platform_issues) == 0
        }

    return results


def check_facebook_compliance(promise):
    """Check Facebook advertising policies."""
    issues = []
    warnings = []
    statement = promise['statement'].lower()

    # Personal attributes policy
    personal_attribute_patterns = [
        'your weight', 'you are overweight', 'your fat',
        'you are struggling', 'your debt', 'your credit'
    ]
    for pattern in personal_attribute_patterns:
        if pattern in statement:
            issues.append({
                'issue': 'PERSONAL_ATTRIBUTES',
                'description': f"Facebook prohibits asserting personal attributes: '{pattern}'",
                'action': 'Reframe to aspirational language'
            })

    # Before/after imagery references
    if 'before' in statement and 'after' in statement:
        warnings.append({
            'warning': 'BEFORE_AFTER',
            'description': 'Before/after claims face stricter review on Facebook',
            'action': 'Consider alternative framing'
        })

    # Sensational language
    sensational = ['shocking', 'miracle', 'secret', 'exposed', 'they don\'t want you to know']
    for term in sensational:
        if term in statement:
            warnings.append({
                'warning': 'SENSATIONAL_LANGUAGE',
                'description': f"'{term}' may trigger ad rejection",
                'action': 'Use less sensational alternative'
            })

    return issues, warnings


def check_google_compliance(promise):
    """Check Google Ads policies."""
    issues = []
    warnings = []
    statement = promise['statement'].lower()

    # Unrealistic claims
    unrealistic = ['guaranteed results', 'miracle', 'instant', 'overnight success']
    for term in unrealistic:
        if term in statement:
            issues.append({
                'issue': 'UNREALISTIC_CLAIMS',
                'description': f"Google prohibits unrealistic claims: '{term}'",
                'action': 'Remove or moderate claim'
            })

    # Weight loss restrictions
    weight_terms = ['lose weight', 'fat', 'pounds', 'slim']
    if any(term in statement for term in weight_terms):
        warnings.append({
            'warning': 'WEIGHT_LOSS_RESTRICTIONS',
            'description': 'Weight loss ads face additional review and restrictions',
            'action': 'Ensure landing page complies with weight loss policies'
        })

    return issues, warnings
```

### Step 5: Generate Compliance Summary

```python
def generate_compliance_summary(promise, all_checks):
    """
    Generate overall compliance summary and recommendations.
    """
    total_issues = []
    total_warnings = []
    required_disclaimers = []
    compliant = True

    for check_name, check_result in all_checks.items():
        if check_result.get('issues'):
            total_issues.extend(check_result['issues'])
            compliant = False

        if check_result.get('warnings'):
            total_warnings.extend(check_result['warnings'])

        if check_result.get('required_disclaimers'):
            required_disclaimers.extend(check_result['required_disclaimers'])

    # Calculate compliance score
    score = 10 - (len(total_issues) * 3) - (len(total_warnings) * 0.5)

    # Determine status
    if compliant and not total_warnings:
        status = 'COMPLIANT'
    elif compliant:
        status = 'COMPLIANT_WITH_WARNINGS'
    elif len(total_issues) <= 2:
        status = 'FIXABLE'
    else:
        status = 'HIGH_RISK'

    return {
        'promise_id': promise['promise_id'],
        'compliance_score': max(0, round(score, 1)),
        'compliance_status': status,
        'total_issues': total_issues,
        'total_warnings': total_warnings,
        'required_disclaimers': required_disclaimers,
        'recommendations': generate_compliance_fixes(total_issues, total_warnings)
    }
```

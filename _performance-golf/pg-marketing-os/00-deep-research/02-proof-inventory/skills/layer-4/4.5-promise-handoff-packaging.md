# 4.5 Promise Handoff Packaging

## Purpose
Package proof inventory findings specifically for the 04-Promise Skill. The Promise Skill needs to know what promises can be credibly made based on available proof. This microskill creates a tailored handoff that enables promise calibration.

---

## Input Requirements

From Layer 2:
- Promise ceiling calculation
- Category strength scores
- Gap analysis

From Layer 4:
- Knockout proof
- All ranked elements

From Context:
- Schwartz stage
- Intended promises (if pre-defined)

---

## Promise Skill Needs

```
WHAT THE PROMISE SKILL NEEDS TO KNOW:

1. PROMISE CEILING
   └── What's the maximum credible promise given proof strength?

2. SUPPORTED CLAIMS
   └── Which specific claims have proof backing?

3. UNSUPPORTED CLAIMS
   └── Which claims should NOT be made?

4. PROOF FOR PROMISE VARIANTS
   └── What proof supports bold vs. safe promise versions?

5. CONSTRAINT DOCUMENTATION
   └── Why can't we claim certain things?

6. UPLIFT POTENTIAL
   └── How could promise ceiling increase with more proof?
```

## Execution Protocol

### Step 1: Promise Ceiling Packaging

```python
def package_promise_ceiling(ceiling_data, gaps):
    """
    Package promise ceiling data for Promise Skill.
    """

    ceiling_package = {
        'level': ceiling_data['level'],
        'level_name': ceiling_data['name'],
        'description': ceiling_data['description'],

        'allowed_language': [],
        'forbidden_language': [],
        'constraints': [],
        'constraint_rationales': []
    }

    # Language mapping by level
    LEVEL_LANGUAGE = {
        5: {
            'allowed': ['transformation', 'complete', 'total', 'dramatic',
                       'life-changing', 'revolutionary', 'breakthrough'],
            'tone': 'bold'
        },
        4: {
            'allowed': ['significant', 'major', 'real', 'substantial',
                       'measurable', 'proven', 'powerful'],
            'tone': 'confident'
        },
        3: {
            'allowed': ['noticeable', 'meaningful', 'clear', 'positive',
                       'genuine', 'effective', 'helpful'],
            'tone': 'moderate'
        },
        2: {
            'allowed': ['may help', 'could support', 'some users report',
                       'potential', 'possible', 'could contribute'],
            'tone': 'hedged'
        },
        1: {
            'allowed': ['consider', 'explore', 'learn about',
                       'worth trying', 'an approach'],
            'tone': 'cautious'
        }
    }

    current_level = ceiling_data['level']

    # Set allowed language
    level_info = LEVEL_LANGUAGE.get(current_level, LEVEL_LANGUAGE[3])
    ceiling_package['allowed_language'] = level_info['allowed']
    ceiling_package['tone'] = level_info['tone']

    # Set forbidden language (all levels above current)
    forbidden = []
    for level in range(current_level + 1, 6):
        forbidden.extend(LEVEL_LANGUAGE.get(level, {}).get('allowed', []))
    ceiling_package['forbidden_language'] = forbidden

    # Document constraints
    for gap in gaps:
        if gap['severity_tier'] in ['critical', 'high']:
            constraint = {
                'type': gap['gap_type'],
                'category': gap.get('category', ''),
                'impact': f"Cannot make strong {gap.get('category', '')} claims",
                'rationale': gap.get('description', ''),
                'resolution': gap.get('recommendation', '')
            }
            ceiling_package['constraints'].append(constraint)
            ceiling_package['constraint_rationales'].append(
                f"{gap.get('category', '').upper()} GAP: {gap.get('description', '')}"
            )

    return ceiling_package
```

### Step 2: Claim-Proof Mapping

```python
def map_claims_to_proof(elements, intended_claims=None):
    """
    Map specific claims to the proof that supports them.
    """

    # Standard claim types
    CLAIM_TYPES = [
        {
            'claim_type': 'result_claim',
            'example': 'Lose X pounds in Y weeks',
            'proof_needed': ['testimonial_specific_result', 'data_success_rate', 'study_clinical_trial']
        },
        {
            'claim_type': 'transformation_claim',
            'example': 'Transform your life/health/body',
            'proof_needed': ['testimonial_transformation', 'demo_before_after_photo', 'case_study']
        },
        {
            'claim_type': 'mechanism_claim',
            'example': 'Works because of [mechanism]',
            'proof_needed': ['mechanism_study', 'mechanism_expert_explanation', 'study_peer_reviewed']
        },
        {
            'claim_type': 'safety_claim',
            'example': 'Safe and natural',
            'proof_needed': ['certification', 'study_safety', 'endorsement_medical']
        },
        {
            'claim_type': 'superiority_claim',
            'example': 'Better than alternatives',
            'proof_needed': ['data_comparison', 'study_comparative', 'demo_comparison']
        },
        {
            'claim_type': 'speed_claim',
            'example': 'Results in X days/weeks',
            'proof_needed': ['testimonial_specific_result', 'data_time_figures', 'study_clinical_trial']
        },
        {
            'claim_type': 'risk_free_claim',
            'example': 'Risk-free to try',
            'proof_needed': ['guarantee_unconditional', 'trial_free', 'guarantee_double']
        }
    ]

    claim_mapping = []

    for claim_spec in CLAIM_TYPES:
        # Find supporting proof
        supporting_proof = []
        for element in elements:
            if element.sub_type in claim_spec['proof_needed']:
                supporting_proof.append({
                    'element_id': element.id,
                    'sub_type': element.sub_type,
                    'strength': element.stage_adjusted_score,
                    'raw_text': element.raw_text[:200]  # Preview
                })

        # Sort by strength
        supporting_proof.sort(key=lambda x: x['strength'], reverse=True)

        # Determine claim supportability
        if supporting_proof and supporting_proof[0]['strength'] >= 7:
            status = 'strongly_supported'
        elif supporting_proof and supporting_proof[0]['strength'] >= 5:
            status = 'moderately_supported'
        elif supporting_proof:
            status = 'weakly_supported'
        else:
            status = 'unsupported'

        claim_mapping.append({
            'claim_type': claim_spec['claim_type'],
            'example': claim_spec['example'],
            'status': status,
            'supporting_proof': supporting_proof[:3],  # Top 3
            'recommendation': get_claim_recommendation(status, claim_spec['claim_type'])
        })

    return claim_mapping


def get_claim_recommendation(status, claim_type):
    """
    Generate recommendation for making this type of claim.
    """

    RECOMMENDATIONS = {
        'strongly_supported': 'Can make this claim confidently with proof citation',
        'moderately_supported': 'Can make this claim with hedging language',
        'weakly_supported': 'Make claim cautiously, consider adding "may" or "could"',
        'unsupported': 'Do not make this claim—no proof support'
    }

    return RECOMMENDATIONS.get(status, 'Assess carefully before claiming')
```

### Step 3: Promise Variant Support

```python
def package_variant_support(elements, ceiling):
    """
    Package proof support for bold vs. safe promise variants.
    """

    variants = {
        'bold_version': {
            'supported': ceiling['level'] >= 4,
            'proof_requirements_met': [],
            'proof_requirements_missing': [],
            'use_when': 'High-proof situations, warm traffic, Stage 1-2 markets'
        },
        'safe_version': {
            'supported': True,  # Always have a safe version
            'proof_used': [],
            'hedging_language': [],
            'use_when': 'Cold traffic, skeptical markets, Stage 4-5'
        },
        'specific_version': {
            'supported': False,
            'specific_numbers': [],
            'proof_source': [],
            'use_when': 'When specific numbers can be cited'
        }
    }

    # Check bold version requirements
    if ceiling['level'] >= 4:
        variants['bold_version']['proof_requirements_met'].append('Overall proof strength sufficient')

        # Check for transformation proof
        transformation_proof = [e for e in elements if e.sub_type == 'testimonial_transformation']
        if transformation_proof:
            variants['bold_version']['proof_requirements_met'].append(
                f"Transformation testimonial available ({transformation_proof[0].id})"
            )
        else:
            variants['bold_version']['proof_requirements_missing'].append('Missing transformation testimonial')

    # Package safe version
    hedging = ['may help', 'could support', 'many users report', 'has been shown to']
    variants['safe_version']['hedging_language'] = hedging

    # Check for specific numbers
    specific_elements = [e for e in elements
                        if e.scores.specificity >= 8
                        and e.sub_type in ['testimonial_specific_result', 'data_exact_statistic']]

    if specific_elements:
        variants['specific_version']['supported'] = True
        for elem in specific_elements[:3]:
            variants['specific_version']['specific_numbers'].append({
                'element_id': elem.id,
                'number_found': extract_numbers(elem.raw_text),
                'context': elem.raw_text[:100]
            })

    return variants
```

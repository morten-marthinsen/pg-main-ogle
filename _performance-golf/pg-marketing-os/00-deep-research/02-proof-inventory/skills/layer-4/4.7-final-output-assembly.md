## Output Schema Validation

```python
def validate_output_schema(output):
    """
    Validate the complete output against required schema.
    """

    REQUIRED_SECTIONS = [
        'summary',
        'by_category',
        'gaps',
        'rankings',
        'handoffs',
        'elements'
    ]

    REQUIRED_SUMMARY_FIELDS = [
        'total_elements',
        'overall_strength_score',
        'promise_ceiling',
        'schwartz_stage_used',
        'knockout_proof_id'
    ]

    REQUIRED_CATEGORY_FIELDS = [
        'count',
        'strength',
        'status',
        'top_elements'
    ]

    REQUIRED_HANDOFF_SECTIONS = [
        'to_promise',
        'to_big_idea',
        'to_proof_demonstration'
    ]

    validation = {
        'is_valid': True,
        'missing_sections': [],
        'missing_fields': [],
        'warnings': []
    }

    # Check required sections
    for section in REQUIRED_SECTIONS:
        if section not in output:
            validation['missing_sections'].append(section)
            validation['is_valid'] = False

    # Check summary fields
    if 'summary' in output:
        for field in REQUIRED_SUMMARY_FIELDS:
            if field not in output['summary']:
                validation['missing_fields'].append(f'summary.{field}')
                validation['is_valid'] = False

    # Check category structure
    if 'by_category' in output:
        for category in ['social', 'authority', 'demonstration',
                        'mechanism', 'data', 'risk_reversal']:
            if category in output['by_category']:
                for field in REQUIRED_CATEGORY_FIELDS:
                    if field not in output['by_category'][category]:
                        validation['missing_fields'].append(
                            f'by_category.{category}.{field}'
                        )

    # Check handoffs
    if 'handoffs' in output:
        for handoff in REQUIRED_HANDOFF_SECTIONS:
            if handoff not in output['handoffs']:
                validation['missing_sections'].append(f'handoffs.{handoff}')
                validation['is_valid'] = False

    return validation
```

```python
def compile_summary(inventory_data, rankings, schwartz_stage):
    """
    Compile the executive summary section.
    """

    summary = {
        'timestamp': datetime.now().isoformat(),
        'skill_version': '1.0',

        # Core metrics
        'total_elements': len(inventory_data['elements']),
        'overall_strength_score': inventory_data['overall_strength'],
        'promise_ceiling': inventory_data['ceiling']['name'],
        'promise_ceiling_level': inventory_data['ceiling']['level'],
        'schwartz_stage_used': schwartz_stage,

        # Knockout identification
        'knockout_proof_id': rankings['knockout']['element'].id if rankings['knockout'] else None,
        'knockout_potential': rankings['knockout']['knockout_potential'] if rankings['knockout'] else None,

        # Category overview
        'strongest_categories': identify_strongest(inventory_data['category_scores']),
        'weakest_categories': identify_weakest(inventory_data['category_scores']),

        # Gap summary
        'total_gaps': len(inventory_data['gaps']),
        'critical_gaps': count_critical_gaps(inventory_data['gaps']),

        # Discovery summary
        'discoveries_added': inventory_data.get('discoveries_added', 0),
        'generation_recommendations': len(inventory_data.get('generation_plan', [])),

        # Quality assessment
        'inventory_readiness': assess_readiness(inventory_data)
    }

    return summary


def assess_readiness(inventory_data):
    """
    Assess overall inventory readiness for copywriting.
    """

    score = inventory_data['overall_strength']
    critical_gaps = count_critical_gaps(inventory_data['gaps'])

    if score >= 70 and critical_gaps == 0:
        return 'ready'
    elif score >= 50 and critical_gaps <= 1:
        return 'ready_with_constraints'
    elif score >= 30:
        return 'needs_attention'
    else:
        return 'insufficient'
```

### Step 2: Compile Category Details

```python
def compile_category_details(elements, category_scores, gaps):
    """
    Compile detailed breakdown by category.
    """

    CATEGORIES = ['social', 'authority', 'demonstration',
                  'mechanism', 'data', 'risk_reversal']

    by_category = {}

    for category in CATEGORIES:
        cat_elements = [e for e in elements if e.category == category]
        cat_score = category_scores.get(category, {})
        cat_gaps = [g for g in gaps if g.get('category') == category]

        # Sort elements by score
        cat_elements.sort(key=lambda e: e.stage_adjusted_score, reverse=True)

        by_category[category] = {
            'count': len(cat_elements),
            'strength': cat_score.get('strength', 0),
            'status': cat_score.get('status', 'unknown'),

            'top_elements': [
                {
                    'id': e.id,
                    'sub_type': e.sub_type,
                    'score': e.stage_adjusted_score,
                    'preview': e.raw_text[:150] + '...' if len(e.raw_text) > 150 else e.raw_text
                }
                for e in cat_elements[:5]
            ],

            'gaps': [
                {
                    'gap_id': g['gap_id'],
                    'type': g['gap_type'],
                    'severity': g['severity_tier'],
                    'description': g.get('description', '')
                }
                for g in cat_gaps
            ],

            'coverage_assessment': assess_category_coverage(cat_elements, cat_gaps)
        }

    return by_category
```

### Step 3: Compile Rankings Section

```python
def compile_rankings(knockout, position_rankings, objection_mapping, gradualization):
    """
    Compile all rankings into unified section.
    """

    rankings = {
        # Knockout proof
        'knockout_proof': {
            'element_id': knockout['element'].id if knockout else None,
            'sub_type': knockout['element'].sub_type if knockout else None,
            'knockout_potential': knockout.get('knockout_potential'),
            'rationale': knockout.get('rationale'),
            'runner_up_id': knockout.get('runner_up', {}).get('id') if knockout else None
        } if knockout else None,

        # Position rankings
        'by_position': {
            'lead': compile_position_list(position_rankings.get('lead', [])),
            'mechanism': compile_position_list(position_rankings.get('mechanism', [])),
            'body': compile_position_list(position_rankings.get('body', [])),
            'close': compile_position_list(position_rankings.get('close', []))
        },

        # Objection mapping
        'by_objection': {
            obj_id: {
                'objection': data['objection'],
                'coverage': data['coverage_status'],
                'proof_sequence': [
                    {'element_id': p['element'].id, 'match_strength': p['match_strength']}
                    for p in data['proof_matches'][:3]
                ]
            }
            for obj_id, data in objection_mapping.items()
        },

        # Gradualization sequence
        'gradualization_sequence': [
            {
                'position': item['position'],
                'element_id': item['element'].id,
                'phase': item['phase'],
                'rationale': item.get('rationale', '')
            }
            for item in gradualization
        ]
    }

    return rankings


def compile_position_list(position_items):
    """
    Compile position ranking list.
    """

    return [
        {
            'element_id': item['element'].id,
            'sub_type': item['element'].sub_type,
            'position_score': item['position_score'],
            'affinity': item.get('affinity'),
            'rationale': f"{item['element'].sub_type} scores {item['position_score']:.1f}"
        }
        for item in position_items[:5]
    ]
```

```python
def compile_elements_list(elements):
    """
    Compile complete list of all proof elements with full details.
    """

    elements_list = []

    for element in elements:
        element_data = {
            'id': element.id,
            'category': element.category,
            'sub_type': element.sub_type,
            'sub_type_id': element.sub_type_id,

            'raw_text': element.raw_text,
            'extracted_components': element.components,

            'scores': {
                'specificity': element.scores.specificity,
                'credibility': element.scores.credibility,
                'relevance': element.scores.relevance,
                'novelty': element.scores.novelty,
                'emotional_impact': element.scores.emotional_impact,
                'composite': element.composite_score,
                'stage_adjusted': element.stage_adjusted_score
            },

            'quality_tier': element.quality_tier,
            'knockout_potential': getattr(element, 'knockout_potential', None),

            'metadata': {
                'source_file': element.metadata.get('source_file'),
                'extraction_method': element.metadata.get('extraction_method'),
                'discovery_source': element.metadata.get('discovery_source')  # If from Layer 3
            }
        }

        elements_list.append(element_data)

    # Sort by stage-adjusted score
    elements_list.sort(key=lambda e: e['scores']['stage_adjusted'], reverse=True)

    return elements_list
```

### Step 6: Final Assembly

```python
def assemble_final_output(all_data):
    """
    Assemble the complete proof-inventory-output.json
    """

    output = {
        # Executive summary
        'summary': compile_summary(
            all_data['inventory'],
            all_data['rankings'],
            all_data['schwartz_stage']
        ),

        # Category breakdown
        'by_category': compile_category_details(
            all_data['elements'],
            all_data['category_scores'],
            all_data['gaps']
        ),

        # Gap analysis
        'gaps': {
            'total': len(all_data['gaps']),
            'by_severity': group_gaps_by_severity(all_data['gaps']),
            'missing_categories': all_data.get('missing_categories', []),
            'weak_categories': all_data.get('weak_categories', []),
            'recommendations': all_data.get('gap_recommendations', []),
            'generation_plan': all_data.get('generation_plan')
        },

        # All rankings
        'rankings': compile_rankings(
            all_data['knockout'],
            all_data['position_rankings'],
            all_data['objection_mapping'],
            all_data['gradualization']
        ),

        # Downstream handoffs
        'handoffs': compile_handoffs(
            all_data['promise_handoff'],
            all_data['big_idea_handoff'],
            all_data['position_rankings'],
            all_data['gradualization']
        ),

        # Complete element list
        'elements': compile_elements_list(all_data['elements'])
    }

    # Validate before returning
    validation = validate_output_schema(output)
    if not validation['is_valid']:
        output['_validation_warnings'] = validation

    return output
```

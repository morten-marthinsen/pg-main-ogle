# 2.7 Promise Ceiling Calculation

## Purpose
Determine the maximum credible promise level based on available proof strength. The promise ceiling is a critical constraint that flows to the Promise skill—it defines what can believably be claimed.

---

## Input Requirements

From 2.4 Overall Strength Calculation:
- Overall strength score (0-100)
- Category strength breakdown

From 2.5/2.6 Gap Analysis:
- Critical gaps identified
- Gap severities

From Context:
- Schwartz stage
- Intended promises (if defined)

---

├── "Meaningful benefits"
├── "Clear improvement"

└── Example: "Start seeing positive changes"

## Execution Protocol

### Step 1: Determine Base Ceiling from Score

```python
def get_base_ceiling(overall_strength):
    if overall_strength >= 80:
        return {
            'level': 5,
            'name': 'transformation',
            'description': 'Can make bold transformation claims',
            'promise_range': [
                'Complete transformation',
                'Total freedom from problem',
                'Life-changing results'
            ]
        }
    elif overall_strength >= 60:
        return {
            'level': 4,
            'name': 'significant_improvement',
            'description': 'Can make strong improvement claims',
            'promise_range': [
                'Major measurable improvement',
                'Significant results',
                'Real change'
            ]
        }
    elif overall_strength >= 40:
        return {
            'level': 3,
            'name': 'noticeable_improvement',
            'description': 'Can make moderate improvement claims',
            'promise_range': [
                'Noticeable improvement',
                'Meaningful difference',
                'Clear benefits'
            ]
        }
    elif overall_strength >= 20:
        return {
            'level': 2,
            'name': 'some_benefit',
            'description': 'Limited to mild benefit claims',
            'promise_range': [
                'May help with',
                'Could support',
                'Some users report'
            ]
        }
    else:
        return {
            'level': 1,
            'name': 'minimal',
            'description': 'Build proof before making claims',
            'promise_range': [
                'An approach to consider',
                'Worth exploring'
            ]
        }
```

### Step 2: Apply Critical Gap Constraints

Critical gaps can lower the ceiling regardless of overall score:

```python
def apply_gap_constraints(base_ceiling, gaps, schwartz_stage):
    ceiling = base_ceiling.copy()
    constraints = []

    # Check for ceiling-lowering gaps
    for gap in gaps:
        if gap['severity_tier'] == 'critical':
            # Critical gaps lower ceiling by 1 level
            ceiling['level'] = max(1, ceiling['level'] - 1)
            constraints.append({
                'type': 'critical_gap',
                'gap': gap['gap_id'],
                'impact': 'Ceiling reduced by 1 level',
                'reason': f"Critical {gap.get('category', '')} gap"
            })

    # Stage-specific ceiling constraints
    stage_constraints = check_stage_constraints(gaps, schwartz_stage)
    constraints.extend(stage_constraints)

    ceiling['constraints'] = constraints
    ceiling['adjusted'] = len(constraints) > 0

    return ceiling


def check_stage_constraints(gaps, stage):
    constraints = []

    STAGE_REQUIREMENTS = {
        3: {'mechanism': 'Cannot claim mechanism superiority without mechanism proof'},
        4: {'authority': 'Cannot claim credibility without strong authority'},
        5: {'social': 'Cannot claim community/identity without strong social proof'}
    }

    if stage in STAGE_REQUIREMENTS:
        for category, message in STAGE_REQUIREMENTS[stage].items():
            cat_gaps = [g for g in gaps if g.get('category') == category and
                       g['severity_tier'] in ['critical', 'high']]
            if cat_gaps:
                constraints.append({
                    'type': 'stage_requirement',
                    'stage': stage,
                    'category': category,
                    'impact': message
                })

    return constraints
```

### Step 3: Generate Specific Constraints

```python
def generate_specific_constraints(category_scores, gaps):
    constraints = []

    # Constraint: No transformation claims without strong social proof
    if category_scores['social']['status'] not in ['robust', 'adequate']:
        constraints.append({
            'constraint': 'no_transformation_claims',
            'reason': 'Social proof insufficient for transformation claims',
            'threshold_needed': 'At least adequate social proof (1+ strong, 2+ medium)'
        })

    # Constraint: No mechanism claims without mechanism proof
    if category_scores['mechanism']['status'] in ['missing', 'weak']:
        constraints.append({
            'constraint': 'no_mechanism_claims',
            'reason': 'Cannot claim mechanism superiority without proof',
            'threshold_needed': 'At least moderate mechanism proof'
        })

    # Constraint: No specific timeframe claims without data
    data_has_time = any(g.get('sub_type') == 'data_time_figures'
                       for g in gaps if g.get('gap_type') != 'subtype_gap')
    if category_scores['data']['status'] in ['missing', 'weak']:
        constraints.append({
            'constraint': 'no_specific_timeframe_claims',
            'reason': 'Data proof insufficient for specific timeframe promises',
            'threshold_needed': 'Time-based statistics with citations'
        })

    return constraints
```

### Step 4: Generate Promise Guidance

```python
def generate_promise_guidance(ceiling, constraints, gaps):
    guidance = {
        'supported_language': [],
        'unsupported_language': [],
        'specific_recommendations': []
    }

    # Supported language based on ceiling level
    level_language = {
        5: ['transformation', 'complete', 'total', 'dramatic', 'life-changing'],
        4: ['significant', 'major', 'real', 'substantial', 'measurable'],
        3: ['noticeable', 'meaningful', 'clear', 'positive'],
        2: ['may help', 'could support', 'some users report'],
        1: ['consider', 'explore', 'learn about']
    }

    guidance['supported_language'] = level_language[ceiling['level']]

    # Unsupported language (above ceiling)
    for level in range(ceiling['level'] + 1, 6):
        guidance['unsupported_language'].extend(level_language.get(level, []))

    # Specific recommendations
    for constraint in constraints:
        if constraint['type'] == 'critical_gap':
            guidance['specific_recommendations'].append(
                f"Fill {constraint.get('category', 'critical')} gap to enable stronger promises"
            )

    return guidance
```

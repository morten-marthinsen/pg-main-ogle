# 4.2 Supporting Promise Selection

## Purpose
Select 3-5 supporting promises that complement the primary promise. Supporting promises address different benefits, objections, and emotional angles while maintaining coherence with the primary thesis.

---

## Input Requirements

From 4.1 Primary Promise Selection:
- Confirmed primary promise
- Primary components (core promise, mechanism, villain)

From 3.7 Validation Assembly:
- All validated promises
- Validation statuses and scores

From Context:
- Campaign scope (single promise vs multi-benefit)
- Copy length (determines how many supporting promises needed)

---

## Execution Protocol

### Step 1: Filter Eligible Supporting Promises

```python
def filter_supporting_candidates(validated_promises, primary):
    """
    Filter promises eligible to be supporting.
    """
    eligible = []

    primary_core = extract_core_promise(primary['statement']).lower()

    for promise in validated_promises:
        # Skip the primary
        if promise['promise_id'] == primary['promise_id']:
            continue

        # Must be validated
        if promise['validation_status']['status'] in ['VALIDATION_FAILED']:
            continue

        # Must not duplicate primary
        promise_core = extract_core_promise(promise['statement']).lower()
        similarity = calculate_similarity(primary_core, promise_core)
        if similarity > 0.8:
            continue  # Too similar

        # Must not contradict primary
        if check_contradiction(promise, primary):
            continue

        eligible.append(promise)

    return eligible
```

### Step 2: Categorize by Role

```python
def categorize_by_role(eligible_promises, primary, context):
    """
    Categorize each promise by its potential role.
    """
    categorized = {
        'benefit_expansion': [],
        'objection_preemption': [],
        'emotional_variety': [],
        'proof_anchors': [],
        'mechanism_reinforcement': []
    }

    primary_type = primary.get('promise_type')
    primary_emotion = primary.get('emotional_frame')
    primary_benefit = extract_benefit_category(primary['statement'])

    for promise in eligible_promises:
        roles = identify_roles(promise, primary_type, primary_emotion, primary_benefit)

        for role in roles:
            categorized[role].append({
                'promise': promise,
                'role_fit': calculate_role_fit(promise, role)
            })

    # Sort each category by role fit
    for role in categorized:
        categorized[role].sort(key=lambda x: x['role_fit'], reverse=True)

    return categorized


def identify_roles(promise, primary_type, primary_emotion, primary_benefit):
    """
    Identify which roles this promise could fill.
    """
    roles = []

    promise_type = promise.get('promise_type')
    promise_emotion = promise.get('emotional_frame')
    promise_benefit = extract_benefit_category(promise['statement'])

    # Benefit expansion
    if promise_benefit != primary_benefit:
        roles.append('benefit_expansion')

    # Emotional variety
    if promise_emotion and promise_emotion != primary_emotion:
        roles.append('emotional_variety')

    # Proof anchor
    if promise.get('validation_scores', {}).get('proof_verification', 0) >= 8:
        roles.append('proof_anchors')

    # Objection preemption
    if has_objection_preemption_language(promise):
        roles.append('objection_preemption')

    # Mechanism reinforcement
    if promise.get('mechanism_connected') and 'root cause' in promise['statement'].lower():
        roles.append('mechanism_reinforcement')

    return roles if roles else ['benefit_expansion']  # Default


def has_objection_preemption_language(promise):
    """Check if promise preempts objections."""
    statement = promise['statement'].lower()
    preemption_patterns = [
        'even if', 'no matter', 'regardless',
        'without', 'don\'t have to', 'finally'
    ]
    return any(p in statement for p in preemption_patterns)
```

### Step 3: Select Optimal Mix

```python
def select_supporting_mix(categorized, primary, num_supporting=4):
    """
    Select optimal mix of supporting promises.
    """
    selected = []
    used_ids = set()
    role_coverage = set()

    # Priority order for role coverage
    priority_roles = [
        'benefit_expansion',
        'emotional_variety',
        'proof_anchors',
        'objection_preemption',
        'mechanism_reinforcement'
    ]

    # First pass: one from each priority role (if available)
    for role in priority_roles:
        if len(selected) >= num_supporting:
            break

        candidates = categorized.get(role, [])
        for candidate in candidates:
            promise = candidate['promise']
            if promise['promise_id'] not in used_ids:
                selected.append({
                    'promise': promise,
                    'primary_role': role,
                    'role_fit': candidate['role_fit']
                })
                used_ids.add(promise['promise_id'])
                role_coverage.add(role)
                break

    # Second pass: fill remaining slots with highest-scoring
    if len(selected) < num_supporting:
        all_remaining = []
        for role, candidates in categorized.items():
            for candidate in candidates:
                if candidate['promise']['promise_id'] not in used_ids:
                    all_remaining.append(candidate)

        all_remaining.sort(key=lambda x: x['promise']['final_score'], reverse=True)

        for candidate in all_remaining:
            if len(selected) >= num_supporting:
                break
            selected.append({
                'promise': candidate['promise'],
                'primary_role': 'additional',
                'role_fit': candidate['role_fit']
            })
            used_ids.add(candidate['promise']['promise_id'])

    return selected, role_coverage
```

### Step 4: Document Supporting Promise Set

```python
def document_supporting_set(selected, primary):
    """
    Create documentation for supporting promise set.
    """
    documentation = {
        'count': len(selected),
        'role_coverage': list(set(s['primary_role'] for s in selected)),
        'supporting_promises': []
    }

    for i, selection in enumerate(selected):
        promise = selection['promise']

        doc = {
            'order': i + 1,
            'promise_id': promise['promise_id'],
            'statement': promise['statement'],
            'primary_role': selection['primary_role'],

            'relationship_to_primary': describe_relationship(promise, primary),

            'scores': {
                'final_score': promise['final_score'],
                'validation_composite': promise.get('validation_composite'),
                'role_fit': selection['role_fit']
            },

            'usage_guidance': generate_supporting_usage(promise, selection['primary_role']),

            'proof_pairing': promise.get('proof_matches', [{}])[0]
        }

        documentation['supporting_promises'].append(doc)

    return documentation


def describe_relationship(supporting, primary):
    """Describe how supporting relates to primary."""
    supporting_benefit = extract_benefit_category(supporting['statement'])
    primary_benefit = extract_benefit_category(primary['statement'])

    supporting_emotion = supporting.get('emotional_frame')
    primary_emotion = primary.get('emotional_frame')

    relationships = []

    if supporting_benefit != primary_benefit:
        relationships.append(f"Expands benefit: {primary_benefit} → {supporting_benefit}")

    if supporting_emotion and supporting_emotion != primary_emotion:
        relationships.append(f"Different emotional angle: {supporting_emotion}")

    if supporting.get('mechanism_connected') and primary.get('mechanism_connected'):
        relationships.append("Reinforces mechanism")

    return relationships if relationships else ["Complementary benefit"]


def generate_supporting_usage(promise, role):
    """Generate usage guidance based on role."""
    usage_by_role = {
        'benefit_expansion': {
            'placement': 'Body copy, after primary benefit established',
            'intro_pattern': '"And that\'s not all..."',
            'proof_approach': 'Stack with testimonial'
        },
        'emotional_variety': {
            'placement': 'Alternate lead option, or body for emotional diversity',
            'intro_pattern': '"But more importantly..."',
            'proof_approach': 'Use emotionally resonant testimonial'
        },
        'proof_anchors': {
            'placement': 'Proof section, demonstration section',
            'intro_pattern': '"Just look at [Name]..."',
            'proof_approach': 'Lead with knockout proof'
        },
        'objection_preemption': {
            'placement': 'After potential objection point',
            'intro_pattern': '"Even if you\'ve tried everything..."',
            'proof_approach': 'Use "tried everything" testimonials'
        },
        'mechanism_reinforcement': {
            'placement': 'Mechanism section transition',
            'intro_pattern': '"Here\'s why this works..."',
            'proof_approach': 'Use scientific proof'
        }
    }

    return usage_by_role.get(role, {
        'placement': 'Body copy',
        'intro_pattern': '"You\'ll also..."',
        'proof_approach': 'Standard testimonial'
    })
```

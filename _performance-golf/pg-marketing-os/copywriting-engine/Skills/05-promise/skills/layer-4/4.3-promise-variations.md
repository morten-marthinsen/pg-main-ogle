# 4.3 Promise Variations

## Purpose
Generate copy-ready variations of all selected promises (primary + supporting). Each promise needs multiple versions for different contexts: headlines, body copy, testimonial setups, closes, emails, and ads.

---

## Input Requirements

From 4.1 Primary Promise Selection:
- Primary promise with initial variations

From 4.2 Supporting Promise Selection:
- Supporting promises with roles

From Context:
- Copy formats needed (VSL, sales letter, email, ads)
- Platform requirements
- Brand voice guidelines

## Execution Protocol

### Step 1: Generate Core Variations

```python
def generate_variations(promise, variation_types, context):
    """
    Generate all variation types for a promise.
    """
    core = extract_core_promise(promise['statement'])
    mechanism = promise.get('mechanism_name', '')
    villain = promise.get('villain', '')
    timeframe = extract_timeframe(promise['statement'])

    variations = {}

    for var_type in variation_types:
        generator = VARIATION_GENERATORS.get(var_type)
        if generator:
            variations[var_type] = generator(core, mechanism, villain, timeframe, context)

    return variations


VARIATION_GENERATORS = {
    'statement': generate_statement_variations,
    'headline': generate_headline_variations,
    'hook': generate_hook_variations,
    'question': generate_question_variations,
    'testimonial_setup': generate_testimonial_variations,
    'future_pace': generate_future_pace_variations,
    'benefit_bullet': generate_bullet_variations,
    'subject_line': generate_subject_lines,
    'ad_copy': generate_ad_variations
}


def generate_statement_variations(core, mechanism, villain, timeframe, context):
    """Generate full statement variations."""
    variations = []

    # With mechanism
    if mechanism:
        variations.append(f"The {mechanism} way to {core}")
        variations.append(f"{core.capitalize()} with the {mechanism}")

    # With villain
    if villain:
        variations.append(f"Defeat the {villain} and {core}")
        variations.append(f"When you defeat the {villain}, you can {core}")

    # With timeframe
    if timeframe:
        variations.append(f"{core.capitalize()} — in {timeframe}")
        variations.append(f"In just {timeframe}, {core}")

    # Clean variations
    variations.append(core.capitalize())

    return variations


def generate_headline_variations(core, mechanism, villain, timeframe, context):
    """Generate headline-optimized versions."""
    variations = []

    # Punchy headline formats
    variations.append(f"The Secret to {core.title()}")
    variations.append(f"How to {core.capitalize()}")
    variations.append(f"Finally: {core.capitalize()}")

    # Number format
    if timeframe:
        variations.append(f"The {timeframe} Formula to {core.capitalize()}")

    # ONLY format
    variations.append(f"The ONLY Way to {core.capitalize()}")

    # Villain format
    if villain:
        variations.append(f"Defeat {villain} and {core.capitalize()}")

    return variations


def generate_hook_variations(core, mechanism, villain, timeframe, context):
    """Generate short hook versions."""
    hooks = []

    # Direct hooks
    hooks.append(f"Finally {core}")
    hooks.append(f"{core.capitalize()} — starting today")

    # Curiosity hooks
    hooks.append(f"The truth about {core.split()[0]}ing")

    # Villain hook
    if villain:
        hooks.append(f"The {villain} secret")

    return hooks


def generate_question_variations(core, mechanism, villain, timeframe, context):
    """Generate question format variations."""
    questions = []

    questions.append(f"What if you could finally {core}?")
    questions.append(f"Ready to {core}?")
    questions.append(f"Want to know the secret to {core.split()[0]}ing?")

    if timeframe:
        questions.append(f"What would you do if you could {core} in {timeframe}?")

    # Yes/no format
    questions.append(f"Can you really {core}?")

    return questions


def generate_testimonial_variations(core, mechanism, villain, timeframe, context):
    """Generate testimonial setup variations."""
    setups = []

    setups.append(f"I finally {core.replace('your', 'my')}...")
    setups.append(f"After years of trying, I {core.replace('your', 'my').replace('finally ', '')}...")

    if timeframe:
        setups.append(f"In just {timeframe}, I {core.replace('your', 'my')}...")

    setups.append(f"I never thought I'd {core.replace('your', 'my').replace('finally ', '')}...")

    return setups


def generate_future_pace_variations(core, mechanism, villain, timeframe, context):
    """Generate future pacing variations."""
    future_paces = []

    future_paces.append(f"Imagine finally being able to {core}...")
    future_paces.append(f"Picture yourself — you've {past_tense(core)}...")
    future_paces.append(f"What would your life look like if you could {core}?")

    if timeframe:
        future_paces.append(f"In {timeframe} from now, you could {core}...")

    return future_paces


def generate_bullet_variations(core, mechanism, villain, timeframe, context):
    """Generate benefit bullet variations."""
    bullets = []

    bullets.append(f"✓ {core.capitalize()}")
    bullets.append(f"✓ {core.capitalize()} — without willpower")
    bullets.append(f"✓ Finally {core}")

    if timeframe:
        bullets.append(f"✓ {core.capitalize()} in as little as {timeframe}")

    return bullets


def generate_subject_lines(core, mechanism, villain, timeframe, context):
    """Generate email subject lines."""
    subject_lines = []

    # Short and punchy
    subject_lines.append(f"Finally: {core}")
    subject_lines.append(f"The secret to {core.split()[0]}ing")
    subject_lines.append(f"Can you {core}? Yes.")

    # Curiosity
    subject_lines.append(f"Why you can't {core} (until now)")
    subject_lines.append(f"[New] How to {core}")

    # Personal
    subject_lines.append(f"I {past_tense(core.replace('your', 'my'))}")
    subject_lines.append(f"What if you could {core}?")

    # Filter by length (< 50 chars)
    return [s for s in subject_lines if len(s) < 50]


def generate_ad_variations(core, mechanism, villain, timeframe, context):
    """Generate platform-specific ad copy."""
    platform = context.get('platform', 'facebook')

    ad_variations = {
        'facebook': {
            'primary_text': f"Finally: {core.capitalize()}. Without the struggle.",
            'headline': f"How to {core.capitalize()}",
            'description': f"Discover the {mechanism} approach"
        },
        'google': {
            'headline_1': f"{core.capitalize()} | {mechanism}",
            'headline_2': f"Finally {core.title()}",
            'description': f"Discover how the {mechanism} helps you {core}"
        },
        'youtube': {
            'headline': f"The Truth About {core.title()}",
            'intro_hook': f"What if you could {core}? Stay tuned..."
        }
    }

    return ad_variations.get(platform, ad_variations['facebook'])
```

### Step 2: Quality Check Variations

```python
def quality_check_variations(variations, promise):
    """
    Check variations for quality and compliance.
    """
    checked = {}

    for var_type, var_list in variations.items():
        checked[var_type] = []

        for variation in var_list:
            check = {
                'text': variation,
                'word_count': len(variation.split()),
                'compliant': check_compliance(variation),
                'natural': check_natural_language(variation),
                'score': 0
            }

            # Score
            score = 7
            if check['compliant']:
                score += 1
            if check['natural']:
                score += 1
            if appropriate_length(variation, var_type):
                score += 1

            check['score'] = score
            checked[var_type].append(check)

        # Sort by score
        checked[var_type].sort(key=lambda x: x['score'], reverse=True)

    return checked


def appropriate_length(text, var_type):
    """Check if length is appropriate for variation type."""
    word_count = len(text.split())
    length_guidelines = {
        'headline': (5, 15),
        'hook': (3, 10),
        'subject_line': (3, 10),
        'statement': (10, 25),
        'question': (5, 15),
        'bullet': (3, 12)
    }
    min_len, max_len = length_guidelines.get(var_type, (5, 20))
    return min_len <= word_count <= max_len
```

## Assembly Protocol

### Step 1: Aggregate All Validation Data

```python
def aggregate_validation_data(promise, validation_results):
    """
    Combine all validation results for a single promise.
    """
    aggregated = {
        'promise_id': promise['promise_id'],
        'statement': promise['statement'],
        'rank': promise['rank'],
        'tier': promise['tier'],
        'calibration_score': promise.get('calibration_composite'),

        'validation_scores': {
            'proof_verification': validation_results['proof']['verification']['strength'],
            'objection_resilience': validation_results['objection']['resilience_score'],
            'believability': validation_results['believability']['believability_score'],
            'compliance': validation_results['compliance']['compliance_score'],
            'customer_voice': validation_results['voice']['voice_score'],
            'story_coherence': validation_results['coherence']['coherence_score']
        },

        'validation_statuses': {
            'proof': validation_results['proof']['verification']['level'],
            'objection': validation_results['objection']['resilience_status'],
            'believability': validation_results['believability']['believability_status'],
            'compliance': validation_results['compliance']['compliance_status'],
            'voice': validation_results['voice']['voice_status'],
            'coherence': validation_results['coherence']['coherence_status']
        },

        'issues': collect_all_issues(validation_results),
        'required_actions': collect_required_actions(validation_results),
        'disclaimers_required': validation_results['compliance'].get('required_disclaimers', [])
    }

    return aggregated


def collect_all_issues(results):
    """Collect issues from all validation checks."""
    issues = []

    # Proof issues
    if results['proof']['verification'].get('level') in ['verified_weak', 'unverified']:
        issues.append({
            'source': 'proof',
            'issue': 'Proof insufficient',
            'severity': 'high'
        })

    # Objection issues
    for concern in results['objection'].get('concerning', []):
        issues.append({
            'source': 'objection',
            'issue': f"Objection unanswered: {concern['objection']}",
            'severity': 'medium' if concern['intensity'] < 8 else 'high'
        })

    # Believability issues
    if results['believability']['believability_score'] < 6:
        issues.append({
            'source': 'believability',
            'issue': 'Low believability',
            'severity': 'high'
        })

    # Compliance issues
    for issue in results['compliance'].get('total_issues', []):
        issues.append({
            'source': 'compliance',
            'issue': issue.get('description', issue.get('issue')),
            'severity': 'high'
        })

    # Voice issues
    if results['voice']['voice_score'] < 5:
        issues.append({
            'source': 'voice',
            'issue': 'Doesn\'t sound like customer',
            'severity': 'medium'
        })

    # Coherence issues
    for issue in results['coherence'].get('issues', []):
        issues.append({
            'source': 'coherence',
            'issue': issue,
            'severity': 'medium'
        })

    return issues
```

### Step 2: Calculate Validation Composite

```python
def calculate_validation_composite(aggregated):
    """
    Calculate overall validation score.
    """
    weights = {
        'proof_verification': 0.25,
        'objection_resilience': 0.15,
        'believability': 0.20,
        'compliance': 0.20,
        'customer_voice': 0.10,
        'story_coherence': 0.10
    }

    scores = aggregated['validation_scores']

    # Normalize all scores to 0-10
    normalized = {dim: min(10, max(0, scores[dim])) for dim in weights}

    composite = sum(normalized[dim] * weights[dim] for dim in weights)

    return round(composite, 2)
```

### Step 3: Determine Validation Status

```python
def determine_validation_status(aggregated, composite):
    """
    Assign final validation status.
    """
    scores = aggregated['validation_scores']
    issues = aggregated['issues']

    # Count blocking issues
    high_severity = [i for i in issues if i['severity'] == 'high']

    # Check for minimum thresholds
    below_threshold = []

    if scores['proof_verification'] < 4:
        below_threshold.append('proof')

    if scores['compliance'] < 5:
        below_threshold.append('compliance')

    if scores['believability'] < 4:
        below_threshold.append('believability')

    # Determine status
    if below_threshold:
        return {
            'status': 'VALIDATION_FAILED',
            'blocking_factors': below_threshold,
            'action': 'Must address blocking factors before use'
        }

    if high_severity:
        if len(high_severity) >= 2:
            return {
                'status': 'CONDITIONAL_PASS',
                'issues_to_resolve': len(high_severity),
                'action': 'Address high-severity issues in copy'
            }
        else:
            return {
                'status': 'VALIDATED_WITH_ISSUES',
                'issues_to_resolve': 1,
                'action': 'Minor adjustments needed'
            }

    if composite >= 8:
        return {
            'status': 'FULLY_VALIDATED',
            'action': 'Ready for packaging'
        }

    if composite >= 6.5:
        return {
            'status': 'VALIDATED',
            'action': 'Ready for packaging'
        }

    return {
        'status': 'VALIDATED_MARGINAL',
        'action': 'Use as secondary promise only'
    }
```

### Step 4: Generate Validation Summary

```python
def generate_validation_summary(all_validated):
    """
    Generate summary statistics for validation layer.
    """
    summary = {
        'total_validated': len(all_validated),
        'status_distribution': {
            'fully_validated': 0,
            'validated': 0,
            'validated_with_issues': 0,
            'conditional_pass': 0,
            'validation_failed': 0
        },
        'dimension_averages': {},
        'common_issues': [],
        'disclaimers_needed': []
    }

    # Count by status
    for promise in all_validated:
        status = promise['validation_status']['status']
        if status == 'FULLY_VALIDATED':
            summary['status_distribution']['fully_validated'] += 1
        elif status == 'VALIDATED':
            summary['status_distribution']['validated'] += 1
        elif status == 'VALIDATED_WITH_ISSUES':
            summary['status_distribution']['validated_with_issues'] += 1
        elif status == 'CONDITIONAL_PASS':
            summary['status_distribution']['conditional_pass'] += 1
        else:
            summary['status_distribution']['validation_failed'] += 1

    # Calculate dimension averages
    dimensions = ['proof_verification', 'objection_resilience', 'believability',
                  'compliance', 'customer_voice', 'story_coherence']

    for dim in dimensions:
        scores = [p['validation_scores'][dim] for p in all_validated]
        summary['dimension_averages'][dim] = round(sum(scores) / len(scores), 2)

    # Collect common issues
    all_issues = []
    for promise in all_validated:
        all_issues.extend([i['issue'] for i in promise['issues']])

    issue_counts = {}
    for issue in all_issues:
        issue_counts[issue] = issue_counts.get(issue, 0) + 1

    summary['common_issues'] = sorted(
        [{'issue': k, 'count': v} for k, v in issue_counts.items()],
        key=lambda x: x['count'],
        reverse=True
    )[:5]

    # Collect disclaimers
    all_disclaimers = set()
    for promise in all_validated:
        for disc in promise.get('disclaimers_required', []):
            all_disclaimers.add(disc.get('disclaimer') or disc.get('text'))

    summary['disclaimers_needed'] = list(all_disclaimers)

    return summary
```

### Step 5: Rank Validated Promises

```python
def rank_validated_promises(all_validated):
    """
    Final ranking based on validation + calibration scores.
    """
    for promise in all_validated:
        # Combined score: 60% calibration, 40% validation
        calibration = promise.get('calibration_score', 5)
        validation = promise['validation_composite']

        promise['final_score'] = round(calibration * 0.6 + validation * 0.4, 2)

    # Sort by final score
    all_validated.sort(key=lambda x: x['final_score'], reverse=True)

    # Assign final ranks
    for i, promise in enumerate(all_validated):
        promise['final_rank'] = i + 1

        # Assign final tier
        if promise['validation_status']['status'] == 'VALIDATION_FAILED':
            promise['final_tier'] = 'DISQUALIFIED'
        elif i == 0:
            promise['final_tier'] = 'PRIMARY'
        elif i < 5:
            promise['final_tier'] = 'SUPPORTING'
        elif i < 10:
            promise['final_tier'] = 'SECONDARY'
        else:
            promise['final_tier'] = 'RESERVE'

    return all_validated
```

## Layer 3 Gate Criteria

```yaml
layer_3_gate:

  PASS:
    - primary_promise_status: "FULLY_VALIDATED" or "VALIDATED"
    - supporting_promises_validated >= 3
    - no_primary_tier_validation_failed: true
    - compliance_issues_addressed: true

  PASS_WITH_WARNINGS:
    - PASS criteria met
    - BUT: 2+ promises have "CONDITIONAL_PASS"
    - OR: common_issues_count >= 5

  FAIL:
    - primary_promise_validation_failed: true
    - OR supporting_promises_validated < 2
    - OR blocking_compliance_issues: true
```

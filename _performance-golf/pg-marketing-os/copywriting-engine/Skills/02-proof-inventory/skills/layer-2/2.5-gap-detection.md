# 2.5 Gap Detection

## Purpose
Identify all proof gaps across the inventory—missing categories, weak categories, under-represented sub-types, and unsupported claims. Gap detection drives Discovery and Generation operations.

---

## Input Requirements

From 2.3 Category Strength Scoring:
- Category strength scores and status
- Quality distributions
- Sub-type breakdowns

From Context:
- Promise/claims to be made (if available)
- Mechanism to prove (if defined)
- Schwartz stage

---

## Gap Types

```
GAP TAXONOMY:

1. MISSING CATEGORY
   └── Zero elements in category
   └── Most severe gap type

2. WEAK CATEGORY
   └── All elements score < 4
   └── Present but unusable

3. UNDER-REPRESENTED CATEGORY
   └── Fewer than 3 elements
   └── Limited proof diversity

4. QUALITY GAP
   └── No strong (7+) elements
   └── Missing "knockout" proof

5. SUB-TYPE GAP
   └── Important sub-type missing
   └── e.g., No video testimonials

6. CLAIM GAP
   └── Specific claim lacks supporting proof
   └── Promise unsupported
```

---

## Execution Protocol

### Step 1: Identify Missing Categories

```python
def find_missing_categories(category_scores):
    missing = []
    for cat, scores in category_scores.items():
        if scores['count'] == 0:
            missing.append({
                'category': cat,
                'gap_type': 'missing',
                'severity': 'critical'
            })
    return missing
```

### Step 2: Identify Weak Categories

```python
def find_weak_categories(category_scores):
    weak = []
    for cat, scores in category_scores.items():
        if scores['count'] > 0:
            # All elements are weak (none strong or medium)
            dist = scores['quality_distribution']
            if dist['strong'] == 0 and dist['medium'] == 0:
                weak.append({
                    'category': cat,
                    'gap_type': 'weak',
                    'severity': 'high',
                    'element_count': scores['count'],
                    'issue': 'All elements score below 4'
                })
    return weak
```

### Step 3: Identify Under-Represented Categories

```python
def find_under_represented(category_scores, min_threshold=3):
    under = []
    for cat, scores in category_scores.items():
        count = scores['count']
        if 0 < count < min_threshold:
            under.append({
                'category': cat,
                'gap_type': 'under_represented',
                'severity': 'medium',
                'current_count': count,
                'recommended_count': min_threshold,
                'issue': f'Only {count} elements, recommend {min_threshold}+'
            })
    return under
```

### Step 4: Identify Quality Gaps

```python
def find_quality_gaps(category_scores):
    quality_gaps = []
    for cat, scores in category_scores.items():
        if scores['count'] > 0:
            dist = scores['quality_distribution']
            if dist['strong'] == 0:
                quality_gaps.append({
                    'category': cat,
                    'gap_type': 'quality_gap',
                    'severity': 'medium',
                    'issue': 'No strong (7+) elements - missing knockout proof',
                    'has_medium': dist['medium'],
                    'has_weak': dist['weak']
                })
    return quality_gaps
```

### Step 5: Identify Sub-Type Gaps

Check for important missing sub-types:

```python
IMPORTANT_SUB_TYPES = {
    'social': [
        'testimonial_specific_result',  # Numbers-based
        'testimonial_video',            # Video format
    ],
    'authority': [
        'study_peer_reviewed',          # Highest credibility
        'endorsement_expert',           # Expert backing
    ],
    'demonstration': [
        'before_after_photo',           # Visual proof
        'demo_video',                   # Video demonstration
    ],
    'mechanism': [
        'mechanism_study',              # Research-backed mechanism
        'mechanism_expert_explanation', # Expert explanation
    ],
    'data': [
        'data_source_cited',            # Attributed stats
        'data_exact_statistic',         # Precise numbers
    ],
    'risk_reversal': [
        'guarantee_extended',           # Long guarantee
        'guarantee_unconditional',      # Strong guarantee
    ]
}

def find_subtype_gaps(elements, category_scores):
    subtype_gaps = []

    for cat, important_subtypes in IMPORTANT_SUB_TYPES.items():
        if category_scores[cat]['count'] == 0:
            continue  # Already flagged as missing

        present_subtypes = set(
            e.sub_type for e in elements if e.category == cat
        )

        for subtype in important_subtypes:
            if subtype not in present_subtypes:
                subtype_gaps.append({
                    'category': cat,
                    'gap_type': 'subtype_gap',
                    'missing_subtype': subtype,
                    'severity': 'low',
                    'issue': f'Missing important sub-type: {subtype}'
                })

    return subtype_gaps
```

### Step 6: Identify Claim Gaps

If promise/mechanism context is provided:

```python
def find_claim_gaps(elements, claims_to_support):
    claim_gaps = []

    for claim in claims_to_support:
        supporting_proof = find_supporting_proof(elements, claim)

        if len(supporting_proof) == 0:
            claim_gaps.append({
                'gap_type': 'claim_gap',
                'claim': claim,
                'severity': 'high',
                'issue': f'No proof supports claim: "{claim}"',
                'supporting_elements': []
            })
        elif all(e.adjusted_score < 7 for e in supporting_proof):
            claim_gaps.append({
                'gap_type': 'weak_claim_support',
                'claim': claim,
                'severity': 'medium',
                'issue': f'No strong proof for claim: "{claim}"',
                'supporting_elements': [e.id for e in supporting_proof]
            })

    return claim_gaps
```

- [ ] Important sub-types checked
- [ ] Claim gaps identified (if claims provided)
- [ ] Severity assigned to all gaps
- [ ] Recommendations generated

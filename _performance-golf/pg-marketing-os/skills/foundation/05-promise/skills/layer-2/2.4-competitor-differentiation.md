# 2.4 Competitor Differentiation

## Purpose
Ensure promises are differentiated from competitor claims. In sophisticated markets (Stage 3+), "me too" promises fail. This microskill verifies our promises stand apart from what prospects have already heard.

## Execution Protocol

### Step 1: Map Competitor Promise Landscape

```python
def map_competitor_promises(competitor_data):
    """
    Create comprehensive map of competitor claims.
    """
    promise_landscape = {
        'exact_claims': [],      # Word-for-word similar
        'category_claims': [],   # Same benefit category
        'mechanism_claims': [],  # Claims about HOW
        'specificity_levels': {} # How specific are competitors?
    }

    for competitor in competitor_data:
        name = competitor['name']
        promises = competitor.get('promises', [])

        for promise in promises:
            # Categorize the promise
            category = categorize_promise(promise)
            specificity = assess_specificity(promise)

            promise_landscape['category_claims'].append({
                'competitor': name,
                'promise': promise,
                'category': category,
                'specificity': specificity
            })

            # Track specificity by category
            if category not in promise_landscape['specificity_levels']:
                promise_landscape['specificity_levels'][category] = []
            promise_landscape['specificity_levels'][category].append(specificity)

        # Track mechanism claims
        if competitor.get('mechanism_claim'):
            promise_landscape['mechanism_claims'].append({
                'competitor': name,
                'mechanism': competitor['mechanism_claim']
            })

    return promise_landscape


def categorize_promise(promise):
    """
    Categorize promise into benefit categories.
    """
    categories = {
        'weight_loss': ['lose weight', 'fat', 'pounds', 'slim', 'thin'],
        'energy': ['energy', 'fatigue', 'tired', 'vitality'],
        'cravings': ['craving', 'hunger', 'appetite', 'urge'],
        'health': ['health', 'wellness', 'body', 'immune'],
        'appearance': ['look', 'younger', 'skin', 'glow'],
        'confidence': ['confident', 'self-esteem', 'proud']
    }

    promise_lower = promise.lower()
    for category, keywords in categories.items():
        if any(kw in promise_lower for kw in keywords):
            return category

    return 'other'
```

### Step 2: Assess Each Promise Against Landscape

```python
def assess_differentiation(promise, landscape, mechanism_data, schwartz_stage):
    """
    Assess how differentiated a promise is from competitors.
    """
    assessment = {
        'promise_id': promise['id'],
        'promise': promise['statement'],
        'differentiation_level': None,
        'differentiation_score': 0,
        'competitor_overlap': [],
        'differentiation_factors': [],
        'vulnerability': None,
        'recommendation': None
    }

    promise_category = categorize_promise(promise['statement'])
    promise_specificity = assess_specificity(promise['statement'])

    # Check for exact matches
    for comp_promise in landscape['category_claims']:
        similarity = calculate_similarity(promise['statement'], comp_promise['promise'])

        if similarity > 0.8:
            assessment['competitor_overlap'].append({
                'competitor': comp_promise['competitor'],
                'their_promise': comp_promise['promise'],
                'similarity': similarity,
                'overlap_type': 'near_exact'
            })
        elif similarity > 0.5 and comp_promise['category'] == promise_category:
            assessment['competitor_overlap'].append({
                'competitor': comp_promise['competitor'],
                'their_promise': comp_promise['promise'],
                'similarity': similarity,
                'overlap_type': 'category_overlap'
            })

    # Determine differentiation level
    if not assessment['competitor_overlap']:
        assessment['differentiation_level'] = 'unique_claim'
        assessment['differentiation_score'] = 10
        assessment['differentiation_factors'].append('No competitor making this claim')

    elif all(o['overlap_type'] == 'category_overlap' for o in assessment['competitor_overlap']):
        # Same category but not exact—check for mechanism differentiation
        if promise.get('mechanism_connected'):
            assessment['differentiation_level'] = 'mechanism_differentiated'
            assessment['differentiation_score'] = 8
            assessment['differentiation_factors'].append(
                f"Unique mechanism: {mechanism_data['mechanism_name']}"
            )
        else:
            # Check specificity
            category_specificities = landscape['specificity_levels'].get(promise_category, [])
            avg_competitor_specificity = sum(category_specificities) / len(category_specificities) if category_specificities else 5

            if promise_specificity > avg_competitor_specificity + 2:
                assessment['differentiation_level'] = 'specificity_differentiated'
                assessment['differentiation_score'] = 6
                assessment['differentiation_factors'].append(
                    f"More specific than competitors ({promise_specificity} vs {avg_competitor_specificity:.1f})"
                )
            else:
                assessment['differentiation_level'] = 'undifferentiated'
                assessment['differentiation_score'] = 3
                assessment['vulnerability'] = 'sounds_like_everyone_else'

    else:
        # Near-exact matches exist
        assessment['differentiation_level'] = 'undifferentiated'
        assessment['differentiation_score'] = 2
        assessment['vulnerability'] = 'competitor_saying_same_thing'

    # Stage-based adjustment
    if schwartz_stage >= 4 and assessment['differentiation_level'] == 'undifferentiated':
        assessment['recommendation'] = 'must_differentiate_for_stage_4'
    elif schwartz_stage >= 3 and assessment['differentiation_level'] not in ['unique_claim', 'mechanism_differentiated']:
        assessment['recommendation'] = 'add_mechanism_wrapper'

    return assessment
```

### Step 3: Generate Differentiation Strategies

```python
def generate_differentiation_strategy(assessment, mechanism_data, landscape):
    """
    For undifferentiated promises, suggest differentiation approaches.
    """
    if assessment['differentiation_score'] >= 7:
        return None  # Already differentiated

    strategies = []
    promise = assessment['promise']

    # Strategy 1: Mechanism wrapping
    strategies.append({
        'strategy': 'mechanism_wrap',
        'description': 'Wrap the promise in mechanism language',
        'example': f"The {mechanism_data['mechanism_name']} way to {extract_core_promise(promise)}",
        'difficulty': 'easy',
        'effectiveness': 'high_for_stage_3+'
    })

    # Strategy 2: Specificity boost
    current_specificity = assess_specificity(promise)
    if current_specificity < 8:
        strategies.append({
            'strategy': 'add_specificity',
            'description': 'Make the promise more specific',
            'example': add_specificity_markers(promise),
            'difficulty': 'easy',
            'effectiveness': 'moderate'
        })

    # Strategy 3: Angle shift
    emotional_angles = ['hope', 'relief', 'freedom', 'control', 'vindication']
    current_angle = detect_emotional_angle(promise)
    unused_angles = [a for a in emotional_angles if a != current_angle]

    strategies.append({
        'strategy': 'angle_shift',
        'description': f'Reframe with different emotional angle',
        'suggested_angles': unused_angles[:2],
        'example': reframe_with_angle(promise, unused_angles[0]),
        'difficulty': 'moderate',
        'effectiveness': 'high_for_stage_4-5'
    })

    # Strategy 4: "Only" positioning
    if mechanism_data.get('uniqueness_claim'):
        strategies.append({
            'strategy': 'only_positioning',
            'description': 'Position as the ONLY way',
            'example': f"The ONLY way to {extract_core_promise(promise)} is with the {mechanism_data['mechanism_name']}",
            'difficulty': 'requires_unique_mechanism',
            'effectiveness': 'very_high'
        })

    # Strategy 5: Villain framing
    if mechanism_data.get('villain'):
        strategies.append({
            'strategy': 'villain_frame',
            'description': 'Frame promise as defeating the villain',
            'example': f"Defeat the {mechanism_data['villain']} and finally {extract_core_promise(promise)}",
            'difficulty': 'easy',
            'effectiveness': 'high'
        })

    return {
        'promise_id': assessment['promise_id'],
        'current_differentiation': assessment['differentiation_level'],
        'strategies': strategies,
        'recommended': strategies[0] if strategies else None
    }
```

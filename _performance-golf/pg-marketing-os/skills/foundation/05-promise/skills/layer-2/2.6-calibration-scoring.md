# 2.6 Calibration Scoring

## Purpose
Apply comprehensive scoring to all promise candidates based on Layer 2 calibration work. This synthesizes proof ceiling, stage calibration, mechanism fit, differentiation, and thesis alignment into final calibration scores.

---

## Input Requirements

From 2.1 Proof Ceiling Application:
- ceiling_compliance for each promise
- ceiling_downgrade_applied

From 2.2 Schwartz Stage Calibration:
- stage_appropriate for each promise
- reframing_applied

From 2.3 Mechanism Fit Verification:
- alignment_strength for each promise
- connection_path verified

From 2.4 Competitor Differentiation:
- differentiation_score for each promise
- differentiation_strategies (if needed)

From 2.5 Campaign Thesis Generation:
- primary_thesis
- thesis_alignment potential

From 1.7 Candidate Assembly:
- Pre-calibration scores

## Execution Protocol

### Step 1: Aggregate Calibration Data

```python
def aggregate_calibration_data(promise, calibration_outputs):
    """
    Pull all calibration results for a single promise.
    """
    aggregated = {
        'promise_id': promise['id'],
        'statement': promise['statement'],
        'pre_calibration_score': promise.get('scores', {}).get('composite', 5),
        'calibration_data': {}
    }

    # From 2.1 Proof Ceiling
    proof_result = calibration_outputs['proof_ceiling'].get(promise['id'], {})
    aggregated['calibration_data']['proof'] = {
        'ceiling_compliance': proof_result.get('compliance_status'),
        'ceiling_level': proof_result.get('ceiling_level'),
        'downgrade_applied': proof_result.get('downgrade_applied', False),
        'proof_strength': proof_result.get('proof_strength', 5)
    }

    # From 2.2 Stage Calibration
    stage_result = calibration_outputs['stage_calibration'].get(promise['id'], {})
    aggregated['calibration_data']['stage'] = {
        'stage_appropriate': stage_result.get('stage_appropriate'),
        'required_framing': stage_result.get('required_framing'),
        'current_framing': stage_result.get('current_framing'),
        'reframed': stage_result.get('reframed')
    }

    # From 2.3 Mechanism Fit
    mech_result = calibration_outputs['mechanism_fit'].get(promise['id'], {})
    aggregated['calibration_data']['mechanism'] = {
        'alignment_type': mech_result.get('alignment_type'),
        'alignment_strength': mech_result.get('alignment_strength', 5),
        'verified': mech_result.get('verification_status') == 'VERIFIED'
    }

    # From 2.4 Differentiation
    diff_result = calibration_outputs['differentiation'].get(promise['id'], {})
    aggregated['calibration_data']['differentiation'] = {
        'level': diff_result.get('differentiation_level'),
        'score': diff_result.get('differentiation_score', 5),
        'strategy_applied': diff_result.get('strategy_applied')
    }

    # From 2.5 Thesis
    thesis_result = calibration_outputs['thesis']
    aggregated['calibration_data']['thesis'] = {
        'is_primary': thesis_result.get('primary_thesis', {}).get('promise_used') == extract_core_promise(promise['statement']),
        'is_runner_up': any(
            r.get('promise_used') == extract_core_promise(promise['statement'])
            for r in thesis_result.get('runners_up', [])
        )
    }

    return aggregated
```

### Step 2: Calculate Dimension Scores

```python
def calculate_dimension_scores(aggregated):
    """
    Convert calibration data into dimension scores.
    """
    data = aggregated['calibration_data']
    scores = {}

    # PROOF COMPLIANCE (0-10)
    proof = data['proof']
    if proof['ceiling_compliance'] == 'under_ceiling':
        scores['proof_compliance'] = 10
    elif proof['ceiling_compliance'] == 'at_ceiling':
        scores['proof_compliance'] = 8
    elif proof['downgrade_applied']:
        scores['proof_compliance'] = 5  # Downgraded but now compliant
    elif proof['ceiling_compliance'] == 'over_ceiling':
        scores['proof_compliance'] = 3
    else:
        scores['proof_compliance'] = proof['proof_strength']

    # STAGE FIT (0-10)
    stage = data['stage']
    if stage['stage_appropriate']:
        scores['stage_fit'] = 9
    elif stage['reframed']:
        scores['stage_fit'] = 7  # Reframed to fit
    elif stage['current_framing'] == stage['required_framing']:
        scores['stage_fit'] = 8
    else:
        scores['stage_fit'] = 4

    # MECHANISM ALIGNMENT (0-10)
    mech = data['mechanism']
    scores['mechanism_alignment'] = mech['alignment_strength']
    if not mech['verified']:
        scores['mechanism_alignment'] = max(2, scores['mechanism_alignment'] - 3)

    # DIFFERENTIATION (0-10)
    diff = data['differentiation']
    scores['differentiation'] = diff['score']
    if diff['strategy_applied']:
        scores['differentiation'] = min(10, scores['differentiation'] + 1)

    # THESIS POTENTIAL (0-10)
    thesis = data['thesis']
    if thesis['is_primary']:
        scores['thesis_potential'] = 10
    elif thesis['is_runner_up']:
        scores['thesis_potential'] = 8
    elif scores['mechanism_alignment'] >= 8 and scores['differentiation'] >= 7:
        scores['thesis_potential'] = 6  # Could work as thesis
    else:
        scores['thesis_potential'] = 4

    return scores
```

### Step 3: Calculate Calibration Composite

```python
def calculate_calibration_composite(dimension_scores):
    """
    Calculate weighted composite calibration score.
    """
    weights = {
        'proof_compliance': 0.25,
        'stage_fit': 0.20,
        'mechanism_alignment': 0.20,
        'differentiation': 0.20,
        'thesis_potential': 0.15
    }

    composite = sum(
        dimension_scores[dim] * weights[dim]
        for dim in weights
    )

    return round(composite, 2)
```

### Step 4: Determine Calibration Status

```python
def determine_calibration_status(composite, dimension_scores):
    """
    Assign calibration status based on scores.
    """

    # Check for blocking issues
    blocking_issues = []

    if dimension_scores['proof_compliance'] < 4:
        blocking_issues.append('proof_insufficient')

    if dimension_scores['mechanism_alignment'] < 4:
        blocking_issues.append('mechanism_misaligned')

    if dimension_scores['differentiation'] < 3:
        blocking_issues.append('undifferentiated')

    # Determine status
    if blocking_issues:
        return {
            'status': 'BLOCKED',
            'issues': blocking_issues,
            'action': 'Address blocking issues before proceeding'
        }

    if composite >= 8.0:
        return {
            'status': 'CALIBRATED - TIER 1',
            'issues': [],
            'action': 'Ready for final ranking'
        }

    if composite >= 6.5:
        return {
            'status': 'CALIBRATED - TIER 2',
            'issues': [],
            'action': 'May need enhancement for primary use'
        }

    if composite >= 5.0:
        return {
            'status': 'CALIBRATED - TIER 3',
            'issues': [],
            'action': 'Secondary/supporting promise only'
        }

    return {
        'status': 'BELOW THRESHOLD',
        'issues': ['composite_too_low'],
        'action': 'Consider removing from candidate pool'
    }
```

### Step 5: Generate Calibration Summary

```python
def generate_calibration_summary(all_candidates):
    """
    Summarize calibration results across all candidates.
    """
    summary = {
        'total_candidates': len(all_candidates),
        'tier_distribution': {
            'tier_1': 0,
            'tier_2': 0,
            'tier_3': 0,
            'blocked': 0,
            'below_threshold': 0
        },
        'dimension_averages': {
            'proof_compliance': 0,
            'stage_fit': 0,
            'mechanism_alignment': 0,
            'differentiation': 0,
            'thesis_potential': 0
        },
        'blocking_issues': [],
        'top_candidates': []
    }

    for candidate in all_candidates:
        status = candidate['calibration_status']['status']

        if 'TIER 1' in status:
            summary['tier_distribution']['tier_1'] += 1
        elif 'TIER 2' in status:
            summary['tier_distribution']['tier_2'] += 1
        elif 'TIER 3' in status:
            summary['tier_distribution']['tier_3'] += 1
        elif 'BLOCKED' in status:
            summary['tier_distribution']['blocked'] += 1
            summary['blocking_issues'].extend(candidate['calibration_status']['issues'])
        else:
            summary['tier_distribution']['below_threshold'] += 1

        # Accumulate for averages
        for dim in summary['dimension_averages']:
            summary['dimension_averages'][dim] += candidate['dimension_scores'].get(dim, 0)

    # Calculate averages
    n = len(all_candidates)
    for dim in summary['dimension_averages']:
        summary['dimension_averages'][dim] = round(summary['dimension_averages'][dim] / n, 2)

    # Top candidates (Tier 1 sorted by composite)
    tier_1 = [c for c in all_candidates if 'TIER 1' in c['calibration_status']['status']]
    tier_1.sort(key=lambda x: x['calibration_composite'], reverse=True)
    summary['top_candidates'] = tier_1[:5]

    return summary
```

# 1.6 Customer Language Mapping

## Purpose
Map promise language to actual words and phrases used by the target customer. When the promise uses their exact language, it creates instant recognition — "they're talking to ME."

**Key Principle:** The best promises don't sound like marketing. They sound like the prospect talking to a friend about what they want.

## Language Mapping Process

### Step 1: Build Customer Language Library

```python
def build_language_library(research_output):
    """
    Compile customer language into searchable categories.
    """
    library = {
        'outcomes': [],      # How they describe what they want
        'problems': [],      # How they describe what's wrong
        'emotions': [],      # How they describe how they feel
        'timeframes': [],    # How they describe time expectations
        'comparisons': [],   # How they compare to past/others
        'actions': [],       # Verbs they use for desired actions
        'qualifiers': []     # Specific descriptors they use
    }

    # From desired outcomes
    for outcome in research_output.get('desired_outcomes', []):
        library['outcomes'].append({
            'phrase': outcome['text'],
            'frequency': outcome.get('frequency', 1),
            'source': outcome.get('source', 'research')
        })

    # From pain language
    for pain in research_output.get('pain_phrases', []):
        library['problems'].append({
            'phrase': pain['text'],
            'frequency': pain.get('frequency', 1),
            'emotional_intensity': pain.get('intensity', 'medium')
        })

    # From verbatim quotes
    for quote in research_output.get('avatar_quotes', []):
        extracted = extract_language_elements(quote)
        for category, phrases in extracted.items():
            library[category].extend(phrases)

    return library
```

### Step 2: Identify Language Mismatches

```python
def identify_mismatches(promise, language_library):
    """
    Find where promise language differs from customer language.
    """
    mismatches = []

    # Check outcomes
    promise_outcome = extract_outcome(promise)
    customer_outcomes = [o['phrase'] for o in language_library['outcomes']]

    similarity = find_closest_match(promise_outcome, customer_outcomes)
    if similarity['score'] < 0.7:
        mismatches.append({
            'type': 'outcome',
            'current': promise_outcome,
            'customer_alternative': similarity['closest_match'],
            'similarity_score': similarity['score']
        })

    # Check qualifiers
    promise_qualifiers = extract_qualifiers(promise)
    customer_qualifiers = [q['phrase'] for q in language_library['qualifiers']]

    for qualifier in promise_qualifiers:
        similarity = find_closest_match(qualifier, customer_qualifiers)
        if similarity['score'] < 0.8:
            mismatches.append({
                'type': 'qualifier',
                'current': qualifier,
                'customer_alternative': similarity['closest_match'],
                'similarity_score': similarity['score']
            })

    return mismatches
```

### Step 3: Apply Customer Language

```python
def apply_customer_language(promise, language_library, mismatches):
    """
    Rewrite promise using customer language.
    """
    customer_version = promise

    for mismatch in mismatches:
        if mismatch['similarity_score'] < 0.7 and mismatch['customer_alternative']:
            # Replace with customer language
            customer_version = customer_version.replace(
                mismatch['current'],
                mismatch['customer_alternative']
            )

    return {
        'original': promise,
        'customer_language_version': customer_version,
        'replacements_made': len([m for m in mismatches if m['customer_alternative']]),
        'customer_language_score': calculate_customer_language_score(customer_version, language_library)
    }
```

## Customer Language Score

```python
def calculate_customer_language_score(promise, language_library):
    """
    Score how well promise uses customer language.
    """
    score = 5  # Base score

    # Check for exact phrase matches
    all_customer_phrases = get_all_phrases(language_library)
    promise_phrases = extract_phrases(promise)

    exact_matches = sum(1 for p in promise_phrases if p.lower() in all_customer_phrases)
    score += exact_matches * 1.5

    # Check for similar phrases (semantic match)
    for phrase in promise_phrases:
        if find_similar(phrase, all_customer_phrases, threshold=0.8):
            score += 0.5

    # Penalty for marketing jargon
    marketing_jargon = ['optimize', 'maximize', 'leverage', 'transform', 'revolutionary']
    jargon_count = sum(1 for j in marketing_jargon if j in promise.lower())
    score -= jargon_count * 1

    # Bonus for conversational tone
    if sounds_conversational(promise):
        score += 1

    return max(1, min(10, round(score, 1)))
```

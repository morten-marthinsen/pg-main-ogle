## Assembly Process

### Step 1: Merge All Variations

```python
def assemble_candidates(layer_1_outputs):
    """
    Merge all promise variations into unified candidate pool.
    """
    candidates = []
    seen_statements = set()

    # Process each microskill output
    sources = [
        ('blue_sky', layer_1_outputs.get('blue_sky', [])),
        ('specific', layer_1_outputs.get('specificity', [])),
        ('typed', layer_1_outputs.get('type_variations', [])),
        ('mechanism', layer_1_outputs.get('mechanism_connected', [])),
        ('emotional', layer_1_outputs.get('emotional_framing', [])),
        ('customer_language', layer_1_outputs.get('customer_language', []))
    ]

    for source_name, promises in sources:
        for promise in promises:
            statement = normalize_statement(promise.get('statement') or promise.get('framed_statement'))

            # Deduplicate
            if statement.lower() not in seen_statements:
                seen_statements.add(statement.lower())

                candidate = {
                    'id': generate_candidate_id(),
                    'statement': statement,
                    'source': source_name,
                    'original_data': promise,
                    'attributes': extract_attributes(promise)
                }
                candidates.append(candidate)

    return candidates
```

### Step 2: Extract and Normalize Attributes

```python
def extract_attributes(promise_data):
    """
    Extract all relevant attributes for scoring.
    """
    attributes = {
        # Source tracking
        'source_type': promise_data.get('source', 'unknown'),

        # Specificity
        'has_quantity': bool(promise_data.get('specificity_markers', {}).get('quantity')),
        'has_timeframe': bool(promise_data.get('specificity_markers', {}).get('timeframe')),
        'has_qualifier': bool(promise_data.get('specificity_markers', {}).get('qualifier')),
        'specificity_score': promise_data.get('specificity_score', 5),

        # Promise type
        'promise_type': promise_data.get('type', 'unknown'),

        # Mechanism connection
        'mechanism_connected': promise_data.get('mechanism_connected', False),
        'mechanism_pattern': promise_data.get('pattern', None),
        'is_campaign_thesis': promise_data.get('pattern') == 'campaign_thesis',

        # Emotional framing
        'emotional_frame': promise_data.get('frame', None),
        'emotional_intensity': promise_data.get('intensity', 'medium'),
        'emotional_resonance_score': promise_data.get('emotional_resonance_score', 5),

        # Customer language
        'customer_language_score': promise_data.get('customer_language_score', 5),
        'uses_verbatim': promise_data.get('uses_verbatim', False),

        # Stage fit
        'best_for_stages': promise_data.get('best_for_stage', []),
        'stage_fit_score': promise_data.get('stage_fit_score', 5),

        # Proof backing
        'proof_backed': promise_data.get('proof_backed', False)
    }

    return attributes
```

### Step 3: Apply Initial Scoring

```python
def apply_initial_scoring(candidate):
    """
    Score candidate before calibration.
    This is a pre-calibration score — calibration will adjust.
    """
    attrs = candidate['attributes']
    scores = {}

    # SPECIFICITY SCORE (1-10)
    specificity = attrs.get('specificity_score', 5)
    if attrs['has_quantity'] and attrs['has_timeframe'] and attrs['has_qualifier']:
        specificity = min(10, specificity + 2)
    scores['specificity'] = specificity

    # EXCITEMENT SCORE (1-10)
    # Blue sky promises score higher, backed down later if needed
    excitement = 5
    if attrs['source_type'] == 'blue_sky':
        excitement = 9
    elif attrs['promise_type'] == 'transformation':
        excitement = 8
    elif attrs['promise_type'] == 'relief':
        excitement = 7
    elif attrs['emotional_intensity'] == 'high':
        excitement += 2
    scores['excitement'] = min(10, excitement)

    # MECHANISM FIT SCORE (1-10)
    mechanism_fit = 5
    if attrs['mechanism_connected']:
        mechanism_fit = 7
    if attrs['is_campaign_thesis']:
        mechanism_fit = 9
    scores['mechanism_fit'] = mechanism_fit

    # CUSTOMER LANGUAGE SCORE (1-10)
    scores['customer_language'] = attrs.get('customer_language_score', 5)

    # EMOTIONAL RESONANCE SCORE (1-10)
    scores['emotional_resonance'] = attrs.get('emotional_resonance_score', 5)

    # COMPOSITE SCORE (pre-calibration)
    weights = {
        'specificity': 0.20,
        'excitement': 0.25,
        'mechanism_fit': 0.20,
        'customer_language': 0.15,
        'emotional_resonance': 0.20
    }

    composite = sum(scores[dim] * weights[dim] for dim in weights)
    scores['composite'] = round(composite, 2)

    return scores
```

### Step 4: Categorize and Rank

```python
def categorize_candidates(candidates):
    """
    Categorize candidates for easier processing in Layer 2.
    """
    categorized = {
        'campaign_thesis_candidates': [],
        'transformation_promises': [],
        'improvement_promises': [],
        'relief_promises': [],
        'capability_promises': [],
        'prevention_promises': [],
        'mechanism_forward': [],
        'blue_sky_only': []
    }

    for candidate in candidates:
        attrs = candidate['attributes']

        if attrs['is_campaign_thesis']:
            categorized['campaign_thesis_candidates'].append(candidate)

        if attrs['mechanism_connected'] and not attrs['is_campaign_thesis']:
            categorized['mechanism_forward'].append(candidate)

        promise_type = attrs.get('promise_type', 'unknown')
        type_key = f"{promise_type}_promises"
        if type_key in categorized:
            categorized[type_key].append(candidate)

        if attrs['source_type'] == 'blue_sky' and not attrs['mechanism_connected']:
            categorized['blue_sky_only'].append(candidate)

    return categorized
```

      - composite_score < 5.0
      - OR missing multiple key attributes

## Layer 1 Gate Criteria

```yaml
gate_criteria:

  PASS:
    - total_candidates >= 15
    - tier_1_ready >= 8
    - campaign_thesis_candidates >= 2
    - at_least_3_promise_types_represented: true
    - at_least_5_mechanism_connected: true
    - average_specificity_score >= 6

  PASS_WITH_WARNINGS:
    - PASS criteria met
    - BUT: customer_language_score < 6 for most candidates
    - OR: emotional_framing_coverage < 50%

  FAIL:
    - total_candidates < 15
    - OR tier_1_ready < 5
    - OR no_campaign_thesis_candidates
    - OR average_specificity_score < 5

  on_fail:
    - Return to earlier microskills for additional generation
    - Review upstream inputs (research, mechanism) for gaps
```

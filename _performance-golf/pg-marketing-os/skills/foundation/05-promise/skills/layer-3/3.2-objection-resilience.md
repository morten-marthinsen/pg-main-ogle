# 3.2 Objection Resilience

## Purpose
Test each ranked promise against likely prospect objections. A great promise that triggers immediate skepticism is worse than a good promise that feels believable. This microskill stress-tests promises before they go into copy.

---

From 3.1 Proof Verification:
- Proof-verified promises
- Proof mapping for each promise

From 01-research:
- Known market objections
- Schwartz stage (determines skepticism level)
- Competitor promises (what prospects have heard before)

From 02-proof-inventory:

## Execution Protocol

### Step 1: Identify Trigger Patterns

```python
def identify_objection_triggers(promise):
    """
    Identify which objection triggers this promise hits.
    """
    triggers = []
    statement = promise['statement'].lower()

    # Skepticism triggers
    skepticism_patterns = {
        'extreme': ['never', '100%', 'always', 'guaranteed', 'impossible'],
        'vague': ['amazing', 'incredible', 'transform', 'change your life'],
        'missing_mechanism': lambda s: not promise.get('mechanism_connected', False)
    }

    for trigger_type, patterns in skepticism_patterns.items():
        if callable(patterns):
            if patterns(statement):
                triggers.append({'category': 'skepticism', 'type': trigger_type})
        elif any(p in statement for p in patterns):
            triggers.append({'category': 'skepticism', 'type': trigger_type})

    # Relevance triggers
    if not has_specificity_markers(promise):
        triggers.append({'category': 'relevance', 'type': 'generic'})

    if not has_identity_markers(promise):
        triggers.append({'category': 'relevance', 'type': 'missing_identity'})

    # Credibility triggers
    if not promise.get('proof_verified'):
        triggers.append({'category': 'credibility', 'type': 'unattributed'})

    # Mechanism triggers
    if promise.get('mechanism_connected') and 'only' in statement:
        triggers.append({'category': 'mechanism', 'type': 'novel_claim'})

    # Effort triggers
    if promise.get('promise_type') == 'transformation':
        if not any(w in statement for w in ['easy', 'simple', 'effortless', 'without']):
            triggers.append({'category': 'effort', 'type': 'missing_ease'})

    return triggers
```

### Step 2: Generate Likely Objections

```python
def generate_objections(promise, triggers, schwartz_stage):
    """
    Generate specific objections this promise will face.
    """
    objections = []

    # Stage-based skepticism multiplier
    skepticism_multiplier = {
        1: 0.5,
        2: 0.7,
        3: 1.0,
        4: 1.3,
        5: 1.5
    }.get(schwartz_stage, 1.0)

    for trigger in triggers:
        category = trigger['category']
        trigger_type = trigger['type']

        base_objections = OBJECTION_TEMPLATES.get(category, {}).get(trigger_type, [])

        for objection in base_objections:
            intensity = calculate_objection_intensity(objection, schwartz_stage, skepticism_multiplier)

            objections.append({
                'objection': objection['text'],
                'category': category,
                'trigger': trigger_type,
                'intensity': intensity,
                'requires_answer': intensity >= 6
            })

    # Sort by intensity
    objections.sort(key=lambda x: x['intensity'], reverse=True)

    return objections

def calculate_objection_intensity(objection, stage, multiplier):
    """Calculate how intensely this objection will be felt."""
    base = objection['base_intensity']
    adjusted = base * multiplier
    return min(10, round(adjusted, 1))
```

### Step 3: Test Resilience

```python
def test_promise_resilience(promise, objections, proof_mapping):
    """
    Test how well the promise withstands objections.
    """
    resilience_scores = []
    answerable = []
    concerning = []

    for objection in objections:
        # Check if proof answers this objection
        has_answer = check_proof_answers_objection(objection, proof_mapping)

        # Check if promise language preempts objection
        preempted = check_language_preempts(promise, objection)

        # Score resilience
        if preempted:
            score = 9  # Language already handles it
            resilience_scores.append(score)
            answerable.append({
                'objection': objection['objection'],
                'answer_type': 'preempted',
                'note': 'Promise language already addresses this'
            })

        elif has_answer:
            score = 7  # Proof can answer it
            resilience_scores.append(score)
            answerable.append({
                'objection': objection['objection'],
                'answer_type': 'proof_backed',
                'proof_id': has_answer['proof_id'],
                'note': f"Use {has_answer['proof_type']} proof"
            })

        else:
            score = 3  # No clear answer
            resilience_scores.append(score)
            if objection['intensity'] >= 6:
                concerning.append({
                    'objection': objection['objection'],
                    'intensity': objection['intensity'],
                    'no_answer': True,
                    'recommendation': suggest_resilience_fix(objection)
                })

    # Overall resilience score
    overall = sum(resilience_scores) / len(resilience_scores) if resilience_scores else 5

    return {
        'resilience_score': round(overall, 2),
        'objections_tested': len(objections),
        'answerable': answerable,
        'concerning': concerning,
        'resilience_status': categorize_resilience(overall, len(concerning))
    }


def categorize_resilience(score, concerning_count):
    """Categorize overall resilience."""
    if score >= 8 and concerning_count == 0:
        return 'HIGHLY_RESILIENT'
    elif score >= 6 and concerning_count <= 1:
        return 'RESILIENT'
    elif score >= 5:
        return 'ADEQUATE'
    elif score >= 4:
        return 'VULNERABLE'
    else:
        return 'HIGH_RISK'


def suggest_resilience_fix(objection):
    """Suggest how to fix vulnerability."""
    fixes = {
        'skepticism': 'Add specific proof immediately after claim',
        'relevance': 'Add "even if" qualifier or identity marker',
        'credibility': 'Add source attribution or expert backing',
        'mechanism': 'Explain novel mechanism with science',
        'effort': 'Add ease qualifier or "without" statement'
    }
    return fixes.get(objection['category'], 'Review copy language')
```

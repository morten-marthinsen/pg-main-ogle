# 4.1 Primary Promise Selection

## Purpose
Confirm and document the primary promise—the single most important promise that anchors the entire campaign. This isn't just the highest-scoring promise; it's the one that best synthesizes mechanism, proof, and market position.

---

## Input Requirements

From 3.7 Validation Assembly:
- All validated promises with final scores
- Validation statuses
- Issues and guidance

From 2.5 Campaign Thesis Generation:
- Primary thesis
- Thesis components

From Context:
- Campaign goals
- Copy format requirements
- Target platforms

---

## Execution Protocol

### Step 1: Confirm Primary from Rankings

```python
def confirm_primary_promise(validated_promises, thesis_data, campaign_context):
    """
    Confirm the primary promise from validated candidates.
    """
    # Get top candidates
    top_candidates = [p for p in validated_promises if p['final_tier'] in ['PRIMARY', 'SUPPORTING']]

    # Score each against primary criteria
    primary_scores = []

    for candidate in top_candidates[:5]:
        score = calculate_primary_fitness(candidate, thesis_data, campaign_context)
        primary_scores.append({
            'promise': candidate,
            'primary_fitness': score
        })

    # Sort by primary fitness
    primary_scores.sort(key=lambda x: x['primary_fitness'], reverse=True)

    # Select primary
    primary = primary_scores[0]['promise']
    runners_up = [p['promise'] for p in primary_scores[1:3]]

    return {
        'primary_promise': primary,
        'runners_up': runners_up,
        'selection_rationale': generate_selection_rationale(primary, thesis_data)
    }


def calculate_primary_fitness(candidate, thesis_data, context):
    """
    Calculate how well this promise fits as the primary.
    """
    score = 0

    # Base scores
    score += candidate['final_score'] * 5  # Weight final score heavily

    # Thesis alignment (big bonus)
    thesis_promise = thesis_data.get('primary_thesis', {}).get('promise_used', '')
    if extract_core_promise(candidate['statement']) == thesis_promise:
        score += 15
    elif candidate['promise_id'] in [r.get('promise_id') for r in thesis_data.get('runners_up', [])]:
        score += 8

    # Mechanism connection
    if candidate.get('mechanism_connected'):
        score += 5

    # Villain connection
    if candidate.get('villain_connection'):
        score += 5

    # Validation strength
    if candidate['validation_status']['status'] == 'FULLY_VALIDATED':
        score += 5
    elif candidate['validation_status']['status'] == 'VALIDATED':
        score += 3

    # Format fit
    copy_format = context.get('copy_format', 'vsl')
    format_fit = assess_format_fit(candidate, copy_format)
    score += format_fit * 3

    # Differentiation
    diff_score = candidate.get('differentiation_score', 5)
    score += diff_score

    return score


def assess_format_fit(candidate, copy_format):
    """Assess how well promise fits the copy format."""
    statement = candidate['statement']
    word_count = len(statement.split())

    if copy_format == 'vsl':
        # VSL prefers punchy, spoken-friendly
        if word_count <= 15:
            return 10
        elif word_count <= 20:
            return 7
        else:
            return 4

    elif copy_format == 'sales_letter':
        # Sales letter can handle more
        if word_count <= 25:
            return 9
        else:
            return 6

    elif copy_format == 'email':
        # Email needs subject-line potential
        if word_count <= 10:
            return 10
        elif word_count <= 15:
            return 7
        else:
            return 3

    return 7  # Default
```

### Step 2: Document Primary Promise

```python
def document_primary_promise(primary, thesis_data, context):
    """
    Create comprehensive documentation for the primary promise.
    """
    documentation = {
        'statement': primary['statement'],
        'promise_id': primary['promise_id'],

        'scores': {
            'calibration': primary.get('calibration_score'),
            'validation': primary.get('validation_composite'),
            'final': primary['final_score'],
            'primary_fitness': primary.get('primary_fitness')
        },

        'components': {
            'core_promise': extract_core_promise(primary['statement']),
            'mechanism': extract_mechanism_reference(primary['statement']),
            'villain': extract_villain_reference(primary['statement']),
            'superiority': extract_superiority_claim(primary['statement'])
        },

        'thesis_alignment': {
            'is_campaign_thesis': check_thesis_match(primary, thesis_data),
            'thesis_statement': thesis_data.get('primary_thesis', {}).get('statement')
        },

        'proof_support': {
            'primary_proof': primary.get('proof_matches', [{}])[0],
            'proof_strength': primary.get('validation_scores', {}).get('proof_verification'),
            'proof_integration': primary.get('proof_integration_guidance')
        },

        'usage_specifications': {
            'format': context.get('copy_format'),
            'platforms': context.get('platforms', []),
            'placement': ['headline', 'thesis statement', 'close reinforcement'],
            'word_count': len(primary['statement'].split())
        },

        'compliance': {
            'status': primary.get('validation_status', {}).get('compliance_status'),
            'disclaimers_required': primary.get('disclaimers_required', []),
            'platform_restrictions': primary.get('platform_restrictions', [])
        }
    }

    return documentation
```

### Step 3: Generate Variations

```python
def generate_primary_variations(primary, context):
    """
    Generate variations of primary promise for different uses.
    """
    core = extract_core_promise(primary['statement'])
    mechanism = primary.get('mechanism_name', '[mechanism]')
    villain = primary.get('villain', '')

    variations = {
        'full_statement': primary['statement'],

        'headline_version': truncate_for_headline(primary['statement']),

        'short_version': create_short_version(core, mechanism),

        'question_version': convert_to_question(core),

        'testimonial_setup': create_testimonial_setup(core),

        'close_version': create_close_version(core, mechanism),

        'subject_line_options': generate_subject_lines(primary)
    }

    return variations


def truncate_for_headline(statement):
    """Create headline-friendly version."""
    # Remove articles and simplify
    words = statement.split()
    if len(words) <= 12:
        return statement

    # Keep first compelling part
    # Usually up to first comma or dash
    if '—' in statement:
        return statement.split('—')[0].strip()
    if ',' in statement:
        return statement.split(',')[0].strip()

    return ' '.join(words[:12]) + '...'


def create_short_version(core, mechanism):
    """Create short punchy version."""
    return f"{core} with the {mechanism}"


def convert_to_question(core):
    """Convert to question format."""
    # Handle common patterns
    if 'end' in core.lower():
        return f"What if you could finally {core}?"
    elif 'lose' in core.lower():
        return f"Ready to {core}?"
    else:
        return f"Want to {core}?"


def create_close_version(core, mechanism):
    """Create version for close section."""
    return f"With the {mechanism}, you can {core} — starting today."


def generate_subject_lines(primary):
    """Generate email subject line options."""
    core = extract_core_promise(primary['statement'])

    return [
        f"The ONLY way to {core}",
        f"Finally: {core}",
        f"How I {core} (and you can too)",
        f"[New] The secret to {core}",
        f"Why you can't {core} (until now)"
    ]
```
